10def _overlay_windows_vcvars(env): 

11 if sys.version_info >= (3, 5): 

12 from distutils._msvccompiler import _get_vc_env 

13 vc_arch = 'x64' if IS_64BIT else 'x86' 

14 vc_env = _get_vc_env(vc_arch) 

15 # Keys in `_get_vc_env` are always lowercase. 

16 # We turn them into uppercase before overlaying vcvars 

17 # because OS environ keys are always uppercase on Windows. 

18 # https://stackoverflow.com/a/7797329 

19 vc_env = {k.upper(): v for k, v in vc_env.items()} 

20 for k, v in env.items(): 

21 uk = k.upper() 

22 if uk not in vc_env: 

23 vc_env[uk] = v 

24 return vc_env 

25 else: 

26 return env
