71# ONNX can't handle constants that are lists of tensors, which can 

72# get generated in constant prop. So we split them back into prim::ListConstructs 

73def _split_tensor_list_constants(g, block): 

74 for node in block.nodes(): 

75 for subblock in node.blocks(): 

76 _split_tensor_list_constants(g, subblock) 

77 if node.kind() == "prim::Constant": 

78 output_type = node.output().type() 

79 if output_type.isSubtypeOf(ListType.ofTensors()): 

80 inputs = [g.create("prim::Constant").t_('value', t) 

81 .insertBefore(node).output() 

82 for t in node['value']] 

83 lc = (g.create("prim::ListConstruct", inputs) 

84 .insertBefore(node) 

85 .output() 

86 .setType(ListType.ofTensors())) 

87 node.output().replaceAllUsesWith(lc)

90def _optimize_graph(graph, operator_export_type, _disable_torch_constant_prop=False, fixed_batch_size=False, params_dict=None): 

91 # Inline everyting 

92 torch._C._jit_pass_inline(graph)

97 torch._C._jit_pass_lint(graph)

99 torch._C._jit_pass_remove_inplace_ops(graph) 

100 # we record now record some ops like ones/zeros 

101 # into a trace where we previously recorded constants 

102 # use constant prop to maintain our current level of onnx support 

103 # without implementing symbolics for all of them 

104 if _disable_torch_constant_prop is False: 

105 torch._C._jit_pass_constant_propagation(graph) 

106 _split_tensor_list_constants(graph, graph) 

107 # run dce to eliminate dead parts of the graph that might have been 

108 # left behind by things like symbolic_override 

109 torch._C._jit_pass_dce(graph) 

110 torch._C._jit_pass_lint(graph)

112 torch._C._jit_pass_canonicalize_ops(graph) 

113 torch._C._jit_pass_lint(graph) 

114 

115 torch._C._jit_pass_peephole(graph, True) 

116 torch._C._jit_pass_lint(graph) 

117 

118 if operator_export_type != OperatorExportTypes.RAW: 

119 torch._C._jit_pass_onnx_prepare_inplace_ops_for_onnx(graph) 

120 

121 # onnx does not support tuples, so try to remove them 

122 torch._C._jit_pass_lower_all_tuples(graph) 

123 torch._C._jit_pass_peephole(graph, True) 

124 torch._C._jit_pass_lint(graph) 

125 

126 # onnx only supports tensors, but 1 / 2 = 0.5 and tensor(1) / tensor(2) = 0 

127 torch._C._jit_pass_prepare_division_for_onnx(graph) 

128 

129 torch._C._jit_pass_onnx_remove_print(graph)

132

134 torch.onnx.symbolic_helper._quantized_ops.clear() 

135 # Unpack quantized weights for conv and linear ops and insert into graph. 

136 torch._C._jit_pass_onnx_unpack_quantized_weights(graph, params_dict)

139 torch._C._jit_pass_onnx_quantization_insert_permutes(graph, params_dict)

142 torch._C._jit_pass_custom_pattern_based_rewrite_graph(""" 

143 graph(%Pi): 

144 %Pq = quantized::nhwc2nchw(%Pi) 

145 %Pr = quantized::nchw2nhwc(%Pq) 

146 return (%Pr)""", """ 

147 graph(%Ri): 

148 return (%Ri)""", graph)

151 torch._C._jit_pass_erase_number_types(graph) 

152

154 torch._C._jit_pass_lint(graph) 

155 

156 torch._C._jit_pass_onnx_scalar_type_analysis(graph) 

157 torch._C._jit_pass_lint(graph) 

158

160 torch._C._jit_pass_onnx_peephole(graph, _export_onnx_opset_version, fixed_batch_size) 

161 torch._C._jit_pass_lint(graph) 

162 

163 # graph is not a valid jit graph anymore because types have been replaced 

164 # (e.g. int with Tensor), so it now contains operators that don't actually 

165 # exist. We can't run normal dead code elimination because it'd fail trying 

166 # to look up if an operator has side effects, but we can run a dead code

174 torch._C._jit_pass_lint(graph)

178# We accept dictionnaries and strings as ONNX inputs, 

179# but they should be only for configuration use. 

180# we detect here if these inputs are modified, and if so 

181# we warn the user that the changes won't take effect in the 

182# traced ONNX graph 

183def warn_on_static_input_change(input_states): 

184 for input, traced_input in zip(input_states[0], input_states[1]): 

185 if isinstance(input, dict): 

186 if list(input.keys()) != list(traced_input.keys()): 

187 warning = "We detected that you are modifying a dictionnary that is an input to your " \ 

188 "model. " \ 

189 "Note that dictionaries are allowed as inputs in ONNX but they should be " \ 

190 "handled with care. " \ 

191 "Usages of dictionaries is not recommended, and should not be used except " \ 

192 "for configuration use. " \ 

193 "Also note that the order and values of the keys must remain the same. " 

194 warnings.warn(warning) 

195 elif isinstance(input, str): 

196 if input != traced_input: 

197 warning = "The model seems to have string inputs/outputs. " \ 

198 "Note that strings will not appear as inputs/outputs of the ONNX graph. " 

199 warnings.warn(warning)

213def _decide_keep_init_as_input(keep_initializers_as_inputs, operator_export_type, 

214 opset_version): 

215 # This method encapsulates the logic to decide whether the initializers in the graph 

216 # should be listed as ONNX graph inputs (i.e., whether to choose ONNX IR v3 or v4). 

217 # If keep_initializers_as_inputs is not specified (None), then we decide whether to keep 

218 # intializers as graph inputs (val_keep_init_as_ip) based on export type. If export type 

219 # is ONNX, then do not keep initializers as input (val_keep_init_as_ip=False). For all other 

220 # export types keep initializers as input (val_keep_init_as_ip=True). 

221 # If keep_initializers_as_inputs is specified, then respect it. Unless opset version <= 8, 

222 # in which case it must be ignored because for opset version <= 8, all initializers MUST be 

223 # part of graph input (only ONNX IR v3 is allowed), i.e. val_keep_init_as_ip=True. 

224 

225 # Special handling is needed for opset version 8 or lower, because irrespective 

226 # of user input for keep_initializers_as_inputs, the graph must follow ONNX IR v3 

227 # semantics, i.e. all intializers must be listed as ONNX graph input. 

228 if opset_version < 9: 

229 if keep_initializers_as_inputs is False: 

230 warnings.warn("Setting 'keep_initializers_as_inputs=False' for opset version" 

231 "8 or lower would lead to an invalid ONNX graph. Therefore, " 

232 "'keep_initializers_as_inputs=False' is ignored during export." 

233 "Exported model will have initialiers as graph inputs (compliant " 

234 " to ONNX IR v3).") 

235 return True # i.e. True == initializers are part of graph input (ONNX IR v3) 

236 val_keep_init_as_ip = True if keep_initializers_as_inputs is None else keep_initializers_as_inputs 

237 if keep_initializers_as_inputs is None and operator_export_type is OperatorExportTypes.ONNX: 

238 val_keep_init_as_ip = False 

239 return val_keep_init_as_ip

265def _trace_and_get_graph_from_model(model, args, training): 

266 

267 # A basic sanity check: make sure the state_dict keys are the same 

268 # before and after running the model. Fail fast! 

269 orig_state_dict_keys = _unique_state_dict(model).keys() 

270 

271 # By default, training=False, which is good because running a model in 

272 # training mode could result in internal buffers getting updated, dropout 

273 # getting applied, etc. If you really know what you're doing, you 

274 # can turn training=True (or None, to preserve whatever the original 

275 # training mode was.) 

276 with set_training(model, training): 

277 trace_graph, torch_out, inputs_states = \ 

278 torch.jit._get_trace_graph(model, args, _force_outplace=False, _return_inputs_states=True) 

279 warn_on_static_input_change(inputs_states) 

280 

281 if orig_state_dict_keys != _unique_state_dict(model).keys(): 

282 raise RuntimeError("state_dict changed after running the tracer; " 

283 "something weird is happening in your model!") 

284 

285 return trace_graph, torch_out

288def _model_to_graph(model, args, verbose=False, training=False, 

289 input_names=None, output_names=None, 

290 operator_export_type=OperatorExportTypes.ONNX, 

291 example_outputs=None, propagate=False, 

292 _retain_param_name=False, do_constant_folding=True, 

293 _disable_torch_constant_prop=False, fixed_batch_size=False): 

294 from torch.onnx.symbolic_helper import _export_onnx_opset_version 

295 # Special case for common case of passing a single Tensor 

296 if isinstance(args, torch.Tensor): 

297 args = (args, ) 

298 

299 if isinstance(example_outputs, torch.Tensor): 

300 example_outputs = [example_outputs] 

301 

302 torch_out = None 

303 

304 if isinstance(model, torch.jit.ScriptModule): 

305 assert example_outputs is not None, "example_outputs must be provided when exporting a ScriptModule" 

306 try: 

307 method_graph, params = torch._C._jit_pass_lower_graph(model.forward.graph, model._c) 

308 in_vars, in_desc = torch.jit._flatten(tuple(args) + tuple(params)) 

309 graph = _propagate_and_assign_input_shapes( 

310 method_graph, tuple(in_vars), False, propagate) 

311 except AttributeError: 

312 raise RuntimeError('\'forward\' method must be a script method') 

313 elif isinstance(model, torch.jit.ScriptFunction): 

314 assert example_outputs is not None, "example_outputs must be provided when exporting a TorchScript ScriptFunction" 

315 method = model 

316 params = () 

317 in_vars, in_desc = torch.jit._flatten(tuple(args)) 

318 graph = _propagate_and_assign_input_shapes( 

319 model.graph, tuple(in_vars), False, propagate) 

320 else: 

321 graph, torch_out = _trace_and_get_graph_from_model(model, args, training) 

322 state_dict = _unique_state_dict(model) 

323 params = list(state_dict.values()) 

324 if _retain_param_name: 

325 graph_inputs = list(graph.inputs()) 

326 user_input_num = len(graph_inputs) - len(state_dict) 

327 param_names = list(state_dict.keys()) 

328 for i, inp in enumerate(graph_inputs): 

329 if i >= user_input_num: 

330 inp.setDebugName(param_names[i - user_input_num]) 

331 

332 input_and_param_names = [val.debugName() for val in graph.inputs()] 

333 param_names = input_and_param_names[len(input_and_param_names) - len(params):] 

334 params_dict = dict(zip(param_names, params)) 

335 

336 graph = _optimize_graph(graph, operator_export_type, 

337 _disable_torch_constant_prop=_disable_torch_constant_prop, 

338 fixed_batch_size=fixed_batch_size, params_dict=params_dict) 

339 

340 if isinstance(model, torch.jit.ScriptModule) or isinstance(model, torch.jit.ScriptFunction): 

341 out_vars, _ = torch.jit._flatten(tuple(example_outputs)) 

342 graph = _assign_output_shapes(graph, out_vars) 

343 

344 # NB: ONNX requires complete information about output types, which might be 

345 # erased by some optimizations, so we need to set it explicitly again. 

346 if torch_out is not None: 

347 output_tensors, _ = torch._C._jit_flatten(torch_out) 

348 for output, tensor in zip(graph.outputs(), output_tensors): 

349 output.inferTypeFrom(tensor) 

350 

351 _set_input_and_output_names(graph, input_names, output_names) 

352 

353 # make sure that the param dict and the graph match each other 

354 flatten_args, _ = torch._C._jit_flatten(args) 

355 assert len(params) + len(flatten_args) == sum(1 for _ in graph.inputs()) 

356 

357 input_and_param_names = [val.debugName() for val in graph.inputs()] 

358 param_names = input_and_param_names[len(input_and_param_names) - len(params):] 

359 params_dict = dict(zip(param_names, params)) 

360 

361 if do_constant_folding and _export_onnx_opset_version in [9, 10, 11]: 

362 params_dict = torch._C._jit_pass_onnx_constant_fold(graph, params_dict, 

363 _export_onnx_opset_version) 

364 torch._C._jit_pass_dce_allow_deleting_nodes_with_side_effects(graph) 

365 

366 # For ONNX opset < 9, constants only have three data types: float16, float, double. 

367 # In this pass transform constants of other data types to float/double + cast operator. 

368 if _export_onnx_opset_version < 9: 

369 torch._C._jit_pass_onnx_cast_all_constant_to_floating(graph) 

370 

371 if verbose: 

372 print(graph) 

373 

374 return graph, params_dict, torch_out

483 operator_export_type, strip_doc_string, val_keep_init_as_ip, custom_opsets, 

484 val_add_node_names) 

485 else: 

486 proto, export_map = graph._export_onnx( 

487 {}, opset_version, dynamic_axes, False, operator_export_type, 

488 strip_doc_string, val_keep_init_as_ip, custom_opsets, val_add_node_names) 

489 

490 if enable_onnx_checker and operator_export_type != OperatorExportTypes.ONNX_ATEN_FALLBACK: 

491 # Only run checker if enabled and we are not using ATEN fallback 

492 _check_onnx_proto(proto) 

493 

494 if export_type == ExportTypes.PROTOBUF_FILE: 

495 assert(len(export_map) == 0) 

496 with torch.serialization._open_file_like(f, 'wb') as opened_file: 

497 opened_file.write(proto) 

498 elif export_type in [ExportTypes.ZIP_ARCHIVE, ExportTypes.COMPRESSED_ZIP_ARCHIVE]: 

499 import zipfile 

500 compression = zipfile.ZIP_DEFLATED \ 

501 if export_type == ExportTypes.COMPRESSED_ZIP_ARCHIVE \ 

502 else zipfile.ZIP_STORED 

503 with zipfile.ZipFile(f, 'w', compression=compression) as z: 

504 z.writestr(ONNX_ARCHIVE_MODEL_PROTO_NAME, proto) 

505 for k, v in export_map.items(): 

506 z.writestr(k, v) 

507 elif export_type == ExportTypes.DIRECTORY: 

508 import os 

509 if os.path.exists(f): 

510 assert(os.path.isdir(f)) 

511 else: 

512 os.makedirs(f) 

513 

514 model_proto_file = os.path.join(f, ONNX_ARCHIVE_MODEL_PROTO_NAME) 

515 with torch.serialization._open_file_like(model_proto_file, 'wb') as opened_file: 

516 opened_file.write(proto) 

517 

518 for k, v in export_map.items(): 

519 weight_proto_file = os.path.join(f, k) 

520 with torch.serialization._open_file_like(weight_proto_file, 'wb') as opened_file: 

521 opened_file.write(v) 

522 else: 

523 raise RuntimeError('Unknown export type') 

524 finally: 

525 assert __IN_ONNX_EXPORT 

526 __IN_ONNX_EXPORT = False 

527 return torch_out

616def _graph_op(g, opname, *raw_args, **kwargs): 

617 r""" 

618 Create an ONNX operator 'opname', taking 'args' as inputs and attributes 

619 'kwargs'; returning the node representing the single output of this operator 

620 (see the `outputs` keyword argument for multi-return nodes). 

621 

622 The set of operators and the inputs/attributes they take 

623 is documented at https://github.com/onnx/onnx/blob/master/docs/Operators.md 

624 

625 This function is monkey-patched onto Graph. 

626 

627 Arguments: 

628 opname (string): The ONNX operator name, e.g., `Abs` or `Add`. 

629 args (Node...): The inputs to the operator; usually provided 

630 as arguments to the `symbolic` definition. 

631 kwargs: The attributes of the ONNX operator, with keys named 

632 according to the following convention: `alpha_f` indicates 

633 the `alpha` attribute with type `f`. The valid type specifiers are 

634 `f` (float), `i` (int), `s` (string) or `t` (Tensor). An attribute 

635 specified with type float accepts either a single float, or a 

636 list of floats (e.g., you would say `dims_i` for a `dims` attribute 

637 that takes a list of integers). 

638 outputs (int, optional): The number of outputs this operator returns; 

639 by default an operator is assumed to return a single output. 

640 If `outputs` is greater than one, this functions returns a tuple 

641 of output `Node`, representing each output of the ONNX operator 

642 in positional. 

643 """ 

644 outputs = kwargs.pop('outputs', 1) 

645 

646 # Filter out None attributes, this can be convenient client side because 

647 # now they can pass through None attributes, and have them not show up 

648 kwargs = dict((k, v) for k, v in kwargs.items() if v is not None) 

649 

650 def const_if_tensor(arg): 

651 if arg is None: 

652 return arg 

653 elif isinstance(arg, torch._C.Value): 

654 return arg 

655 else: 

656 return g.op("Constant", value_z=arg) 

657 

658 args = list(const_if_tensor(arg) for arg in raw_args) 

659 n = g.insertNode(_newNode(g, opname, outputs, *args, **kwargs)) 

660 if outputs == 1: 

661 return n.output() 

662 return tuple(o for o in n.outputs())

674def _run_symbolic_function(g, n, inputs, env, operator_export_type=OperatorExportTypes.ONNX): 

675 # NB: Returning None means the node gets cloned as is into 

676 # the new graph 

677 try: 

678 import torch 

679 from torch.onnx.symbolic_helper import _export_onnx_opset_version as opset_version 

680 import torch.onnx.symbolic_registry as sym_registry 

681 

682 sym_registry.register_version('', opset_version) 

683 if operator_export_type == OperatorExportTypes.ONNX_ATEN_FALLBACK: 

684 import torch.onnx.symbolic_caffe2 

685 torch.onnx.symbolic_caffe2.register_quantized_ops('caffe2', opset_version) 

686 

687 # See Note [Export inplace] 

688 # TODO: I think this is not necessary anymore 

689 if n.kind().endswith('_'): 

690 ns_op_name = n.kind()[:-1] 

691 else: 

692 ns_op_name = n.kind() 

693 ns, op_name = ns_op_name.split("::") 

694 if ns == "onnx": 

695 # Use the original node directly 

696 return None 

697 

698 elif ns == "aten": 

699 is_exportable_aten_op = sym_registry.is_registered_op(op_name, '', opset_version) 

700 is_onnx_aten_export = operator_export_type == OperatorExportTypes.ONNX_ATEN 

701 is_aten_fallback_export = operator_export_type == OperatorExportTypes.ONNX_ATEN_FALLBACK 

702 if is_onnx_aten_export or (not is_exportable_aten_op and is_aten_fallback_export): 

703 # Direct ATen export requested 

704 attrs = {k + "_" + n.kindOf(k)[0]: n[k] for k in n.attributeNames()} 

705 outputs = n.outputsSize() 

706 attrs["outputs"] = outputs 

707 return _graph_at(g, op_name, *inputs, aten=True, **attrs) 

708 

709 else: 

710 # Export it regularly 

711 attrs = {k: n[k] for k in n.attributeNames()} 

712 if not is_exportable_aten_op: 

713 warnings.warn("ONNX export failed on ATen operator {} because " 

714 "torch.onnx.symbolic_opset{}.{} does not exist" 

715 .format(op_name, opset_version, op_name)) 

716 op_fn = sym_registry.get_registered_op(op_name, '', opset_version) 

717 return op_fn(g, *inputs, **attrs) 

718 

719 elif ns == "prim": 

720 if op_name == "Constant" and not n.mustBeNone(): 

721 if n.kindOf("value") == "t": 

722 return g.op("Constant", value_t=n["value"]) 

723 if n.kindOf("value") == "s": 

724 return g.op("Constant", value_s=n["value"]) 

725 elif n.kindOf("value") == "is": 

726 value = torch.stack([torch.tensor(v) for v in n["value"]]) if n["value"] else [] 

727 return g.op("Constant", value_t=value) 

728 elif n.kindOf("value") == "fs": 

729 value = torch.stack([torch.tensor(v) for v in n["value"]]) if n["value"] else [] 

730 return g.op("Constant", value_t=value) 

731 elif n.output().type().kind() == "DeviceObjType": 

732 return None 

733 else: 

734 raise RuntimeError("Unsupported prim::Constant kind: `{}`. Send a bug report.".format( 

735 n.kindOf("value"))) 

736 elif n.mustBeNone() or op_name == "ListConstruct" or op_name == "ListUnpack": 

737 # None is not an ONNX operator; keep it as None 

738 # let the exporter handle finally eliminating these 

739 

740 # For ListConstruct/ListUnpack, it will be erased in the ONNX peephole pass 

741 return None 

742 elif op_name == 'Loop' or op_name == 'If': 

743 new_op_outputs = g.op(op_name, *inputs, outputs=n.outputsSize()) 

744 new_node = new_op_outputs[0].node() if n.outputsSize() > 1 else new_op_outputs.node() 

745 for b in n.blocks(): 

746 new_block = new_node.addBlock() 

747 torch._C._jit_pass_onnx_block(b, new_block, operator_export_type, env) 

748 return new_op_outputs 

749 else: 

750 # TODO: we sould lift prim's symbolic out 

751 symbolic_name = 'prim_' + op_name 

752 is_exportable = sym_registry.is_registered_op(symbolic_name, '', opset_version) 

753 if not is_exportable: 

754 warnings.warn("ONNX export failed on primitive operator {}; please report a bug".format(op_name)) 

755 symbolic_fn = sym_registry.get_registered_op(symbolic_name, '', opset_version) 

756 attrs = {k: n[k] for k in n.attributeNames()} 

757 return symbolic_fn(g, *inputs, **attrs) 

758 

759 elif ns == "quantized": 

760 domain = '' 

761 if operator_export_type == OperatorExportTypes.ONNX_ATEN_FALLBACK: 

762 domain = 'caffe2' 

763 attrs = {k: n[k] for k in n.attributeNames()} 

764 

765 if not sym_registry.is_registered_op(op_name, domain, opset_version): 

766 warnings.warn("ONNX export failed on quantized operator {}::{} because " 

767 "torch.onnx.symbolic_opset{}.{} does not exist. " 

768 .format(ns, op_name, opset_version, op_name)) 

769 op_fn = sym_registry.get_registered_op(op_name, domain, opset_version) 

770 return op_fn(g, *inputs, **attrs) 

771 

772 # custom ops 

773 elif sym_registry.is_registered_version(ns, opset_version): 

774 if not sym_registry.is_registered_op(op_name, ns, opset_version): 

775 warnings.warn("ONNX export failed on custom operator {}::{} because " 

776 "torch.onnx.symbolic_opset{}.{} does not exist. " 

777 "Have you registered your symbolic function with " 

778 "torch.onnx.register_custom_op_symbolic(symbolic_name, symbolic_fn)?" 

779 .format(ns, op_name, opset_version, op_name)) 

780 symbolic_fn = sym_registry.get_registered_op(op_name, ns, opset_version) 

781 attrs = {k: n[k] for k in n.attributeNames()} 

782 return symbolic_fn(g, *inputs, **attrs) 

783 

784 else: 

785 warnings.warn("ONNX export failed on an operator with unrecognized namespace {}::{}; " 

786 "If you are trying to export a custom operator, make sure you registered " 

787 "it with the right domain and version." 

788 "Otherwise please report a bug".format(ns, op_name)) 

789 return None 

790 

791 except TypeError as e: 

792 # Handle the specific case where we didn't successfully dispatch. 

793 # Otherwise, the backtrace will have the clues you need. 

794 e.args = ("{} (occurred when translating {})".format(e.args[0], op_name), ) 

795 raise

869# This helper function ensures dynamic axes argument is following the expected format 

870def _validate_dynamic_axes(dynamic_axes, model, input_names, output_names): 

871 if len(dynamic_axes) == 0: 

872 return 

873 

874 if(hasattr(model, 'graph')): 

875 # Extracting set of valid input/output names that shall be used for dynamic_axes 

876 if (input_names is None) or len(input_names) == 0: 

877 input_names = [x.debugName() for x in model.graph.inputs()] 

878 if (output_names is None) or len(output_names) == 0: 

879 output_names = [y.debugName() for y in model.graph.outputs()] 

880 

881 valid_names = set((input_names or []) + (output_names or [])) 

882 

883 # If dynamic axes are provided as a list rather than dictionary, they should 

884 # first get converted to a dictionary in expected format. If desired axes names 

885 # are not provided for dynamic axes, automatic names shall be generated for 

886 # provided dynamic axes of specified input/output 

887 for key, value in dynamic_axes.items(): 

888 if key not in valid_names: 

889 warnings.warn("Provided key {} for dynamic axes is not a valid input/output name".format(key)) 

890 if isinstance(value, list): 

891 warnings.warn('No names were found for specified dynamic axes of provided input.' 

892 'Automatically generated names will be applied to each dynamic axes of input {}'.format(key)) 

893 

894 value_dict = {} 

895 for i, x in enumerate(value): 

896 if not isinstance(x, int): 

897 raise ValueError("The type of axis index is expected to be an integer") 

898 if x in value_dict: 

899 warnings.warn('Duplicate dynamic axis index {} was provided for input {}.' 

900 .format(x, key)) 

901 else: 

902 value_dict[x] = str(key) + '_dynamic_axes_' + str(i + 1) 

903 dynamic_axes[key] = value_dict
