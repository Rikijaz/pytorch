96class GraphPy(object): 

97 """Helper class to convert torch.nn.Module to GraphDef proto and visualization 

98 with TensorBoard. 

99 

100 GraphDef generation operates in two passes: 

101 

102 In the first pass, all nodes are read and saved to two lists. 

103 One list is for input/output nodes (nodes_io), which only have inbound 

104 or outbound connections, but not both. Another list is for internal 

105 operator nodes (nodes_op). The first pass also saves all scope name 

106 appeared in the nodes in scope_name_appeared list for later processing. 

107 

108 In the second pass, scope names are fully applied to all nodes. 

109 debugNameToScopedName is a mapping from a node's ID to its fully qualified 

110 scope name. e.g. Net1/Linear[0]/1. Unfortunately torch.jit doesn't have 

111 totally correct scope output, so this is nontrivial. The function 

112 populate_namespace_from_OP_to_IO and find_common_root are used to 

113 assign scope name to a node based on the connection between nodes 

114 in a heuristic kind of way. Bookkeeping is done with shallowest_scope_name 

115 and scope_name_appeared. 

116 """ 

117 def __init__(self): 

118 self.nodes_op = [] 

119 self.nodes_io = OrderedDict() 

120 self.unique_name_to_scoped_name = {} 

121 self.shallowest_scope_name = 'default' 

122 self.scope_name_appeared = [] 

123 

124 def append(self, x): 

125 if isinstance(x, NodePyIO): 

126 self.nodes_io[x.debugName] = x 

127 if isinstance(x, NodePyOP): 

128 self.nodes_op.append(x) 

129 

130 def printall(self): 

131 print('all nodes') 

132 for node in self.nodes_op: 

133 print(node) 

134 for key in self.nodes_io: 

135 print(self.nodes_io[key]) 

136 

137 def find_common_root(self): 

138 for fullscope in self.scope_name_appeared: 

139 if fullscope: 

140 self.shallowest_scope_name = fullscope.split('/')[0] 

141 

142 def populate_namespace_from_OP_to_IO(self): 

143 for node in self.nodes_op: 

144 for node_output, outputSize in zip(node.outputs, node.outputstensor_size): 

145 self.scope_name_appeared.append(node.scopeName) 

146 self.nodes_io[node_output] = NodeBase(node_output, 

147 node.inputs, 

148 node.scopeName, 

149 outputSize, 

150 op_type=node.kind, 

151 attributes=node.attributes) 

152 

153 self.find_common_root() 

154 

155 for node in self.nodes_op: 

156 for input_node_id in node.inputs: 

157 self.unique_name_to_scoped_name[input_node_id] = node.scopeName + '/' + input_node_id 

158 

159 for key, node in self.nodes_io.items(): 

160 if type(node) == NodeBase: 

161 self.unique_name_to_scoped_name[key] = node.scope + '/' + node.debugName 

162 if hasattr(node, 'input_or_output'): 

163 self.unique_name_to_scoped_name[key] = node.input_or_output + '/' + node.debugName 

164 

165 if hasattr(node, 'scope') and node.scope is not None: 

166 self.unique_name_to_scoped_name[key] = node.scope + '/' + node.debugName 

167 if node.scope == '' and self.shallowest_scope_name: 

168 self.unique_name_to_scoped_name[node.debugName] = self.shallowest_scope_name + '/' + node.debugName 

169 

170 # replace name 

171 for key, node in self.nodes_io.items(): 

172 self.nodes_io[key].inputs = [self.unique_name_to_scoped_name[node_input_id] for node_input_id in node.inputs] 

173 if node.debugName in self.unique_name_to_scoped_name: 

174 self.nodes_io[key].debugName = self.unique_name_to_scoped_name[node.debugName]

193def parse(graph, trace, args=None, omit_useless_nodes=True): 

194 """This method parses an optimized PyTorch model graph and produces 

195 a list of nodes and node stats for eventual conversion to TensorBoard 

196 protobuf format. 

197 

198 Args: 

199 graph (PyTorch module): The model graph to be parsed. 

200 trace (PyTorch JIT TracedModule): The model trace to be parsed. 

201 args (tuple): input tensor[s] for the model. 

202 omit_useless_nodes (boolean): Whether to remove nodes from the graph. 

203 """ 

204 n_inputs = len(args) 

205 

206 scope = {} 

207 nodes_py = GraphPy() 

208 for node in graph.inputs(): 

209 if omit_useless_nodes: 

210 if len(node.uses()) == 0: # number of user of the node (= number of outputs/ fanout) 

211 continue 

212 

213 if node.type().kind() != CLASSTYPE_KIND: 

214 nodes_py.append(NodePyIO(node, 'input')) 

215 

216 attr_to_scope = dict() 

217 for node in graph.nodes(): 

218 if node.kind() == GETATTR_KIND: 

219 attr_name = node.s('name') 

220 parent = node.input().node() 

221 if parent.kind() == GETATTR_KIND: # If the parent node is not the top-level "self" node 

222 parent_attr_name = parent.s('name') 

223 parent_scope = attr_to_scope[parent_attr_name] 

224 attr_scope = parent_scope.split('/')[-1] 

225 attr_to_scope[attr_name] = '{}/{}.{}'.format(parent_scope, attr_scope, attr_name) 

226 else: 

227 attr_to_scope[attr_name] = '__module.{}'.format(attr_name) 

228 # We don't need classtype nodes; scope will provide this information 

229 if node.output().type().kind() != CLASSTYPE_KIND: 

230 node_py = NodePyOP(node) 

231 node_py.scopeName = attr_to_scope[attr_name] 

232 nodes_py.append(node_py) 

233 else: 

234 nodes_py.append(NodePyOP(node)) 

235 

236 for i, node in enumerate(graph.outputs()): # Create sink nodes for output ops 

237 node_py = NodePyIO(node, 'output') 

238 node_py.debugName = "output.{}".format(i + 1) 

239 node_py.inputs = [node.debugName()] 

240 nodes_py.append(node_py)

242 def parse_traced_name(module): 

243 if isinstance(module, torch.jit.TracedModule): 

244 module_name = module._name 

245 else: 

246 module_name = getattr(module, 'original_name', "Module") 

247 return module_name 

248 

249 alias_to_name = dict() 

250 base_name = parse_traced_name(trace) 

251 for name, module in trace.named_modules(prefix='__module'): 

252 mod_name = parse_traced_name(module) 

253 attr_name = name.split('.')[-1] 

254 alias_to_name[name] = '{}[{}]'.format(mod_name, attr_name) 

255 

256 for node in nodes_py.nodes_op: 

257 module_aliases = node.scopeName.split('/') 

258 replacements = [ 

259 alias_to_name[alias] 

260 if alias in alias_to_name 

261 else alias.split('.')[-1] 

262 for alias in module_aliases 

263 ] 

264 node.scopeName = base_name 

265 if any(replacements): 

266 node.scopeName += '/' + '/'.join(replacements) 

267 

268 nodes_py.populate_namespace_from_OP_to_IO() 

269 return nodes_py.to_proto()

272def graph(model, args, verbose=False): 

273 """ 

274 This method processes a PyTorch model and produces a `GraphDef` proto 

275 that can be logged to TensorBoard. 

276 

277 Args: 

278 model (PyTorch module): The model to be parsed. 

279 args (tuple): input tensor[s] for the model. 

280 verbose (bool): Whether to print out verbose information while 

281 processing. 

282 """ 

283 with torch.onnx.set_training(model, False): # TODO: move outside of torch.onnx? 

284 try: 

285 trace = torch.jit.trace(model, args) 

286 graph = trace.graph 

287 torch._C._jit_pass_inline(graph) 

288 except RuntimeError as e: 

289 print(e) 

290 print('Error occurs, No graph saved') 

291 raise e 

292 

293 if verbose: 

294 print(graph) 

295 list_of_nodes = parse(graph, trace, args) 

296 # We are hardcoding that this was run on CPU even though it might have actually 

297 # run on GPU. Note this is what is shown in TensorBoard and has no bearing 

298 # on actual execution. 

299 # TODO: See if we can extract GPU vs CPU information from the PyTorch model 

300 # and pass it correctly to TensorBoard. 

301 # 

302 # Definition of StepStats and DeviceStepStats can be found at 

303 # https://github.com/tensorflow/tensorboard/blob/master/tensorboard/plugins/graph/tf_graph_common/test/graph-test.ts 

304 # and 

305 # https://github.com/tensorflow/tensorboard/blob/master/tensorboard/compat/proto/step_stats.proto 

306 stepstats = RunMetadata(step_stats=StepStats(dev_stats=[DeviceStepStats(device="/device:CPU:0")])) 

307 return GraphDef(node=list_of_nodes, versions=VersionDef(producer=22)), stepstats
