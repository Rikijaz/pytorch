33def allocate_buffers(engine): 

34 h_input = cuda.pagelocked_empty(trt.volume(engine.get_binding_shape(0)), 

35 dtype=trt.nptype(trt.float32)) 

36 h_output = cuda.pagelocked_empty(trt.volume(engine.get_binding_shape(1)), 

37 dtype=trt.nptype(trt.float32)) 

38 d_input = cuda.mem_alloc(h_input.nbytes) 

39 d_output = cuda.mem_alloc(h_output.nbytes) 

40 stream = cuda.Stream() 

41 return h_input, d_input, h_output, d_output, stream

43def load_normalized_test_case(input_shape, test_image, pagelocked_buffer, normalization_hint): 

44 def normalize_image(image): 

45 c, h, w = input_shape 

46 image_arr = np.asarray(image.resize((w, h), Image.ANTIALIAS)).transpose([2, 0, 1])\ 

47 .astype(trt.nptype(trt.float32)).ravel() 

48 if (normalization_hint == 0): 

49 return (image_arr / 255.0 - 0.45) / 0.225 

50 elif (normalization_hint == 1): 

51 return (image_arr / 256.0 - 0.5) 

52 np.copyto(pagelocked_buffer, normalize_image(Image.open(test_image))) 

53 return test_image

68 def build_engine_onnx(self, model_file): 

69 with trt.Builder(TRT_LOGGER) as builder, builder.create_network(flags = 1) as network, trt.OnnxParser(network, TRT_LOGGER) as parser: 

70 builder.max_workspace_size = 1 << 33 

71 with open(model_file, 'rb') as model: 

72 if not parser.parse(model.read()): 

73 for error in range(parser.num_errors): 

74 self.fail("ERROR: {}".format(parser.get_error(error))) 

75 return builder.build_cuda_engine(network)

77 def _test_model(self, model_name, input_shape = (3, 224, 224), normalization_hint = 0): 

78 

79 model = getattr(models, model_name)(pretrained=True) 

80 

81 shape = (1,) + input_shape 

82 dummy_input = (torch.randn(shape),) 

83 onnx_name = model_name + ".onnx" 

84 

85 torch.onnx.export(model, 

86 dummy_input, 

87 onnx_name, 

88 input_names = [], 

89 output_names = [], 

90 verbose=False, 

91 export_params=True, 

92 opset_version=9) 

93 

94 with self.build_engine_onnx(onnx_name) as engine: 

95 h_input, d_input, h_output, d_output, stream = allocate_buffers(engine) 

96 with engine.create_execution_context() as context: 

97 err_count = 0 

98 for index, f in enumerate(self.image_files): 

99 test_case = load_normalized_test_case(input_shape, f,\ 

100 h_input, normalization_hint) 

101 cuda.memcpy_htod_async(d_input, h_input, stream) 

102 

103 context.execute_async_v2(bindings=[d_input, d_output], 

104 stream_handle=stream.handle) 

105 cuda.memcpy_dtoh_async(h_output, d_output, stream) 

106 stream.synchronize() 

107 

108 amax = np.argmax(h_output) 

109 pred = self.labels[amax] 

110 if "_".join(pred.split()) not in\ 

111 os.path.splitext(os.path.basename(test_case))[0]: 

112 err_count = err_count + 1 

113 self.assertLessEqual(err_count, 1, "Too many recognition errors")
