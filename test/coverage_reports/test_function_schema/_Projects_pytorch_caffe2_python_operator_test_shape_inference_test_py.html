592 def InferTensorRunAndCompare(self, model, expected_uninferred_blobs=None): 

593 ''' 

594 Runs shape inference, and then the model to check 

595 that the inferred shapes agree with the actual ones 

596 

597 'expected_uninferred_blobs' is the list of blobs for which type and 

598 shape cannot be inferred. 

599 ''' 

600 (shapes, types) = workspace.InferShapesAndTypes( 

601 [model.param_init_net, model.net], 

602 ) 

603 

604 # .. Create net 

605 workspace.RunNetOnce(model.param_init_net) 

606 workspace.CreateNet(model.net, True) 

607 workspace.RunNet(model.Proto().name) 

608 

609 # ... and then check the shapes mismatch 

610 correct_shapes = {} 

611 correct_types = {} 

612 for b in workspace.Blobs(): 

613 arr = workspace.FetchBlob(b) 

614 correct_shapes[b] = arr.shape 

615 if type(arr) is np.ndarray: 

616 if arr.dtype == np.dtype('float32'): 

617 correct_types[b] = caffe2_pb2.TensorProto.FLOAT 

618 elif arr.dtype == np.dtype('int32'): 

619 correct_types[b] = caffe2_pb2.TensorProto.INT32 

620 # BYTE 

621 # STRING 

622 elif arr.dtype == np.dtype('bool'): 

623 correct_types[b] = caffe2_pb2.TensorProto.BOOL 

624 elif arr.dtype == np.dtype('uint8'): 

625 correct_types[b] = caffe2_pb2.TensorProto.UINT8 

626 elif arr.dtype == np.dtype('int8'): 

627 correct_types[b] = caffe2_pb2.TensorProto.INT8 

628 elif arr.dtype == np.dtype('uint16'): 

629 correct_types[b] = caffe2_pb2.TensorProto.UINT16 

630 elif arr.dtype == np.dtype('int16'): 

631 correct_types[b] = caffe2_pb2.TensorProto.INT16 

632 elif arr.dtype == np.dtype('int64'): 

633 correct_types[b] = caffe2_pb2.TensorProto.INT64 

634 elif arr.dtype == np.dtype('float16'): 

635 correct_types[b] = caffe2_pb2.TensorProto.FLOAT16 

636 elif arr.dtype == np.dtype('float64'): 

637 correct_types[b] = caffe2_pb2.TensorProto.DOUBLE 

638 else: 

639 correct_types[b] = "unknown {}".format(arr.dtype) 

640 else: 

641 correct_types[b] = str(type(arr)) 

642 

643 if expected_uninferred_blobs is None: 

644 expected_uninferred_blobs = [] 

645 for b in correct_shapes: 

646 # skip blobs for which shape couldn't be inferred 

647 if b in expected_uninferred_blobs: 

648 continue 

649 self.assertTrue( 

650 np.array_equal( 

651 np.array(shapes[b]).astype(np.int32), 

652 np.array(correct_shapes[b]).astype(np.int32) 

653 ), 

654 "Shape {} mismatch: {} vs. correct {}".format( 

655 b, shapes[b], correct_shapes[b] 

656 ) 

657 ) 

658 self.assertFalse( 

659 b not in types and b in correct_types, 

660 "Type for {} not defined".format(b), 

661 ) 

662 self.assertEqual( 

663 types[b], 

664 correct_types[b], 

665 "Type {} mismatch: {} vs. {}".format( 

666 b, types[b], correct_types[b], 

667 ) 

668 )
