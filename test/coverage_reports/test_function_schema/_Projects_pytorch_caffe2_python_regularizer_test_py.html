128 """ 

129 1. create a weight blob 

130 2. create random group splits 

131 3. run group_l1_nrom with the weight blob 

132 4. run equivalent np operations to calculate group l1 norm 

133 5. compare if the results from 3 and 4 are equal 

134 """ 

135 def compare_reference(weight, group_boundaries, reg_lambda, output): 

136 group_splits = np.hsplit(weight, group_boundaries[1:-1]) 

137 l2_reg = np.sqrt([np.sum(np.square(g)) for g in group_splits]) 

138 l2_normalized = np.multiply(l2_reg, 

139 np.array([np.sqrt(g.shape[1]) for g in group_splits])) 

140 result = np.multiply(np.sum(l2_normalized), reg_lambda) 

141 npt.assert_almost_equal(result, workspace.blobs[output], decimal=2) 

142 

143 weight = np.random.rand(output_dim, input_num).astype(np.float32) 

144 

145 feature_num = np.random.randint(low=1, high=input_num - 1) 

146 group_boundaries = [0] 

147 group_boundaries = np.append( 

148 group_boundaries, 

149 np.sort( 

150 np.random.choice(range(1, input_num - 1), feature_num, replace=False) 

151 ), 

152 ) 

153 group_boundaries = np.append(group_boundaries, [input_num]) 

154 split_info = np.diff(group_boundaries)
