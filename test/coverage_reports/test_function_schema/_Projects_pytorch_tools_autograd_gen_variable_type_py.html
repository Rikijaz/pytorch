1# Generates VariableType.h/cpp 

2# 

3# VariableType is a subclass of at::Type that provides the binding code 

4# necessary to provide a differentiable version of ATen operators. There are a 

5# number of different things we could mean: 

6# 

7# - Given a non-differentiable forward implementation, we might 

8# directly associate it with a backward implementation to make 

9# it differentiable. This is the common case. 

10# 

11# - Some functions don't need a backwards implementation, because 

12# backpropagation will never propagate beyond them. There are a 

13# number of different reasons why this may be the case: 

14# 

15# - The function has no differentiable inputs 

16# - The function's output is not differentiable 

17# - The function has no data dependency on its input 

18# 

19# - Some function don't need a backwards implementation because they 

20# are implemented as a composition of other (differentiable) ATen 

21# functions. These are dispatched directly to the Type superclass, 

22# which will in turn dispatch back to VariableType for its 

23# differentiable subcomponents. 

24#

49# `torch.jit.trace` have undocumented keyword argument `_force_outplace`, 

50# which force jit to replace functions with outplace variants (for 

51# example `aten::add_` becomes `aten::add`). 

52# 

53# This replacement implemented in-place with minimum modifications of 

54# arguments stack (as it assumes that outplace call has the same arguments 

55# as inplace version). 

56# 

57# However there are no such substitutions available for `aten::fill_` 

58# and `aten::zero_` operators, as we never implemented `aten::fill` 

59# and `aten::zero`. So jit tracing hack replacing `aten::zero_` with 

60# `aten::zeros_like` and replacing `aten::fill_` with `aten::full_like`. 

61# 

62# But as they potentially can have different arguments, we also have 

63# to hack into the stack and add missing ones. 

64# 

65# A possible alternative would be: 

66# 

67# - Add `aten::fill` and `aten::zero` 

68# 

69# - Or keep `aten::zeros_like` arguments aligned with `aten::zero_` 

70# arguments (inside of the `native_functions.yaml`)

327def should_trace(declaration): 

328 # Operations involving Storage or Type are not traceable at the moment 

329 if any(arg['simple_type'] in {'Storage', 'Type', 'ConstQuantizerPtr'} for arg in declaration['arguments']): 

330 return False 

331 # We can't trace functions which don't have any Tensor or TensorList returns 

332 if 'Tensor' not in declaration['return_type']: 

333 return False 

334 name = declaration['name'] 

335 base_name = name[:-1] if declaration['inplace'] else name[:-4] if name.endswith('_out') else name 

336 if base_name in DONT_RECORD_TRACE or name in DONT_RECORD_TRACE: 

337 return False 

338 return True

345def format_postrecord_trace(declaration): 

346 # For outplacing ops, *_out overloads require special handling to move the 

347 # output *argument* to a return value 

348 if is_out_overload(declaration): 

349 output_names_outplace = [arg['name'] for arg in declaration['arguments'] if arg.get('output', False)] 

350 output_names_inplace = [r['name'] for r in declaration['returns']] 

351 

352 # Code size optimization: the common case is that the return value is 

353 # the same for both variants 

354 if output_names_outplace == output_names_inplace: 

355 outputs = ['jit::tracer::addOutput(node, {});'.format(n) for n in output_names_outplace] 

356 return POST_RECORD_TRACE.substitute(add_trace_outputs=outputs) 

357 

358 local = {} 

359 local['cond'] = 'force_outplace' 

360 local['true'] = ['jit::tracer::addOutput(node, {});'.format(n) for n in output_names_outplace] 

361 local['false'] = ['jit::tracer::addOutput(node, {});'.format(n) for n in output_names_inplace] 

362 selection = SELECT.substitute(local) 

363 return POST_RECORD_TRACE.substitute(add_trace_outputs=selection) 

364 

365 output_names = [r['name'] for r in declaration['returns']] 

366 outputs = ['jit::tracer::addOutput(node, {});'.format(n) for n in output_names] 

367 return POST_RECORD_TRACE.substitute(add_trace_outputs=outputs)

370def format_trace_op_name(declaration): 

371 is_inplace = declaration['api_name'] != uninplace_api_name(declaration['api_name']) 

372 

373 if not is_inplace or is_out_overload(declaration): 

374 # special case for *_out functions: the in-place and out-of-place ops 

375 # are overloaded with the same name in the JIT 

376 trace_name = uninplace_api_name(declaration['api_name']) 

377 trace_name = RENAME_TRACE.get(trace_name, trace_name) 

378 return OP_NAME.substitute(trace_name=trace_name) 

379 

380 # otherwise, this is an in-place op and we need to emit both in- and 

381 # out-of-place versions 

382 outplace_trace_name = uninplace_api_name(declaration['api_name']) 

383 inplace_trace_name = declaration['api_name'] 

384 outplace_trace_name = RENAME_TRACE.get(outplace_trace_name, outplace_trace_name) 

385 inplace_trace_name = RENAME_TRACE.get(inplace_trace_name, inplace_trace_name) 

386 

387 select_params = {} 

388 select_params['cond'] = 'tracer_state->force_outplace' 

389 select_params['true'] = OP_NAME.substitute(trace_name=outplace_trace_name) 

390 select_params['false'] = OP_NAME.substitute(trace_name=inplace_trace_name) 

391 

392 return SELECT.substitute(select_params)

395def format_trace_inputs(declaration): 

396 def dispatch_trace_input(arg_spec): 

397 name, value, simple_type, nullable = arg_spec 

398 # XXX: For arg that have type of Tensor?[], tracer will pass allow_undefined to addInputs 

399 if simple_type == 'TensorList' and nullable: 

400 return '''jit::tracer::addInputs(node, "{}", {}, {});'''.format(name, value, "true") 

401 else: 

402 return ADD_TRACE_INPUT.substitute(name=name, input=value) 

403 

404 trace_inputs = declaration['arguments'] 

405 

406 if is_out_overload(declaration): 

407 # *_out functions take the result as a first argument, but they are the 

408 # last argument in the JIT schema. 

409 out_input = trace_inputs[0] 

410 trace_inputs = trace_inputs[1:] 

411 

412 trace_input_spec = [(i['name'], i['name'], i['simple_type'], i.get('is_nullable')) for i in trace_inputs] 

413 

414 trace_inputs = \ 

415 '\n'.join(dispatch_trace_input(arg_spec) for arg_spec in trace_input_spec) 

416 

417 if is_out_overload(declaration): 

418 # for *_out functions, handle the result argument differently for inplace/outplace. 

419 # For inplace: just add the input to the end to confirm with the JIT schema 

420 inplace = ADD_TRACE_INPUT.substitute(name=out_input['name'], input=out_input['name']) 

421 

422 # for outplace: do nothing, except if the declaration is a factory. 

423 # Factories are a bit special because their out-of-place overloads 

424 # take an extra TensorOptions argument, which is missing in the _out function 

425 trace_name = uninplace_api_name(declaration['api_name']) 

426 has_factory_name = trace_name in FACTORY_FUNCTION_NAMES 

427 if has_factory_name: 

428 outplace = ADD_TRACE_INPUT.substitute(name='out', input='out.options()') 

429 else: 

430 outplace = '' 

431 

432 trace_inputs += '\n' 

433 trace_inputs += SELECT.substitute( 

434 cond='tracer_state->force_outplace', true=outplace, false=inplace) 

435 

436 return trace_inputs

439def format_prerecord_trace(declaration): 

440 local = {} 

441 is_inplace = declaration['api_name'] != uninplace_api_name(declaration['api_name']) 

442 

443 local['set_op_name'] = format_trace_op_name(declaration) 

444 

445 is_inplace = declaration['api_name'] != uninplace_api_name(declaration['api_name']) 

446 add_args = '' 

447 if is_inplace: 

448 api_name = uninplace_api_name(declaration['api_name']) 

449 add_args = RENAME_TRACE_ADD_ARGS.get(api_name, '') 

450 if add_args: 

451 select_params = {} 

452 select_params['cond'] = 'tracer_state->force_outplace' 

453 select_params['true'] = add_args 

454 select_params['false'] = '' 

455 additional_inputs = SELECT.substitute(select_params) 

456 else: 

457 additional_inputs = '' 

458 local['add_trace_inputs'] = format_trace_inputs(declaration) + additional_inputs 

459 

460 local['inplace_guard'] = '' 

461 if is_inplace: 

462 local['inplace_guard'] = INPLACE_GUARD.substitute( 

463 name=declaration['api_name'], 

464 mutable_input=declaration['arguments'][0]['name']) 

465 

466 return PRE_RECORD_TRACE.substitute(local)

475def gen_variable_type(out, aten_declarations, template_path): 

476 """VariableType.h and VariableType.cpp body 

477 

478 This is the at::Type subclass for differentiable tensors. The 

479 implementation of each function dispatches to the base tensor type to 

480 compute the output. The grad_fn is attached to differentiable functions. 

481 """ 

482 

483 # WARNING: this function call modifies global mutable state 

484 find_factory_functions(aten_declarations) 

485 

486 aten_declarations = list(sorted(aten_declarations, key=lambda decl: decl['name'])) 

487 

488 gen_variable_type_shard(out, aten_declarations, template_path, None, True) 

489 

490 # NOTE: see Note [Sharded File] at the top of the VariableType.cpp 

491 # template regarding sharding of the generated files. 

492 num_shards = 5 

493 shards = [[] for _ in range(num_shards)] 

494 

495 # functions are assigned arbitrarily but stably to a file based on hash 

496 for decl in aten_declarations: 

497 x = sum(ord(c) for c in decl['name']) % num_shards 

498 shards[x].append(decl) 

499 

500 for i, shard in enumerate(shards): 

501 gen_variable_type_shard(out, shard, template_path, '_%d' % i, False) 

502 gen_variable_type_shard(out, aten_declarations, template_path, 'Everything', False) 

503 

504 REGISTRATION_DECLARATIONS_H = CodeTemplate.from_file(template_path + "/RegistrationDeclarations.h") 

505 registration_declarations = [] 

506 

507 # TODO(Ailing): copy_ and einsum will be removed in followup PRs. 

508 for declaration in aten_declarations: 

509 if dispatch_strategy(declaration) == 'use_derived' or declaration['name'] in ('copy_', 'resize_', 'einsum'): 

510 registration_declarations.append(REGISTRATION_DECLARATION.substitute(declaration)) 

511 

512 env = { 

513 'registration_declarations': registration_declarations, 

514 } 

515 write(out, 'RegistrationDeclarations.h', REGISTRATION_DECLARATIONS_H, env)

517def gen_variable_type_shard(out, aten_declarations, template_path, suffix, header): 

518 VARIABLE_TYPE_H = CodeTemplate.from_file(template_path + '/VariableType.h') 

519 VARIABLE_TYPE_CPP = CodeTemplate.from_file(template_path + '/VariableType.cpp') 

520 

521 type_declarations = [] 

522 type_definitions = [] 

523 wrapper_registrations = [] 

524 

525 for declaration in aten_declarations: 

526 formal_types = [arg['type'] for arg in declaration['arguments']] 

527 type_declarations.append(METHOD_DECLARATION.substitute(declaration)) 

528 if not declaration['manual_kernel_registration']: 

529 body = emit_body(declaration) 

530 type_definitions.append(METHOD_DEFINITION.substitute( 

531 declaration, type_definition_body=body)) 

532 if declaration['use_c10_dispatcher'] == 'full': 

533 wrapper_registrations.append(WRAPPER_REGISTRATION.substitute( 

534 declaration, formal_types=formal_types)) 

535 else: 

536 assert declaration['use_c10_dispatcher'] == 'unboxed_only' 

537 wrapper_registrations.append(UNBOXEDONLY_WRAPPER_REGISTRATION.substitute( 

538 declaration, formal_types=formal_types)) 

539 

540 env = { 

541 'type_derived_method_declarations': type_declarations, 

542 'type_derived_method_definitions': type_definitions, 

543 'wrapper_registrations': wrapper_registrations,

545 if header:

578 def find_args_with_derivatives(differentiable_inputs): 

579 """Find arguments that have derivative definitions"""

610 len(differentiable_inputs) > 0 and len(differentiable_outputs) > 0 and 

611 strategy == 'use_derived')

617 def emit_save_inputs(): 

618 setup = [] 

619 if func is None: 

620 return setup 

621 

622 has_tensorlist_arg = any(arg['type'] == 'TensorList' for arg in func['args_with_derivatives']) 

623 

624 # We don't want to save tensors if we know that they will never be used 

625 # when computing the derivative, so we add guards to those statements 

626 def guard_for(arg): 

627 # It's hard to determine the edge offset if we have TensorLists 

628 if has_tensorlist_arg: 

629 return None 

630 

631 # Empirical evaluation of the cases where we insert those guards in 

632 # backward show that they are somewhat useless. E.g. there's no need 

633 # to guard on some values captured from forward, because they had to 

634 # require_grad if the backward function even gets executed. I don't 

635 # have any good ideas for detecting those cases, so I simply disabled the 

636 # checks. 

637 if 'backward' in func['name']: 

638 return None 

639 

640 # If there's a single derivative we could compute, we already have 

641 # a requires_grad check that is sufficient 

642 if len(func['args_with_derivatives']) <= 1: 

643 return None 

644 

645 # We really only care about trimming down the amount of tensors we save 

646 if arg['type'] != 'Tensor': 

647 return None 

648 

649 # We want to emit simple guards, so we only allow that if checking one 

650 # input is enough to determine whether we need that value 

651 used_in = [d for d in func['derivatives'] if arg in d['saved_inputs']] 

652 assert len(used_in) > 0 

653 if len(used_in) != 1: 

654 return None 

655 derivative = used_in[0] 

656 if len(derivative['var_names']) != 1: 

657 return None 

658 derivative_var_name = derivative['var_names'][0] 

659 

660 # Figure out the offset of the edge that uses this variable 

661 for edge_off, arg in enumerate(func['args_with_derivatives']): 

662 if arg['name'] == derivative_var_name: 

663 break 

664 else: 

665 raise AssertionError() 

666 

667 return 'grad_fn->should_compute_output({})'.format(edge_off) 

668 

669 setup.extend(save_variables(func['saved_inputs'], False, guard_for)) 

670 for arg in func['args_with_derivatives']: 

671 if arg['type'] == 'TensorList': 

672 setup.append("grad_fn->{}_size_ = {}.size();".format(arg['name'], arg['name'])) 

673 

674 return setup

701 body.append(DECLARE_GRAD_FN.substitute(env))

723 def save_variables(saved_variables, is_output, guard_for=lambda name: None): 

724 # assign the saved variables to the generated grad_fn 

725 stmts = [] 

726 for arg in saved_variables: 

727 name = arg['name'] 

728 expr = arg.get('expr', arg['name']) 

729 if arg['type'] == 'Tensor' or (is_output and arg['type'] == 'Scalar'): 

730 name += '_' 

731 var = arg['name'] 

732 if var == 'self' and inplace: 

733 var = 'self.clone()' 

734 assert not is_output 

735 if inplace and is_output: 

736 var = 'self' 

737 is_inplace_view = "as_variable_ref({}).is_view()".format(var) 

738 expr = 'SavedVariable({}, {}, {})'.format(var, str(is_output).lower(), is_inplace_view) 

739 else: 

740 expr = 'SavedVariable({}, {})'.format(var, str(is_output).lower()) 

741 elif arg['type'] == 'TensorList': 

742 name += '_' 

743 expr = 'make_saved_variable_list({})'.format(arg['name']) 

744 elif arg['type'] == 'IntArrayRef': 

745 expr = expr + ".vec()" 

746 guard = guard_for(arg) 

747 if guard is None: 

748 stmts.append('grad_fn->{} = {};'.format(name, expr)) 

749 else: 

750 stmts.append('if ({}) {{'.format(guard)) 

751 stmts.append(' grad_fn->{} = {};'.format(name, expr)) 

752 stmts.append('}') 

753 return stmts

764 def wrap_output(call): 

765 # Returns `wrapped_call` which is a drop-in replacement for `call` 

766 if 'Tensor' not in declaration['return_type']: 

767 return call 

768 elif view_info is not None: 

769 # See NOTE [ Autograd View Variables ] in variable.h for details. 

770 differentiable_output_vars = {r['name'] for r in differentiable_outputs} 

771 

772 if not isinstance(view_info, str): 

773 raise TypeError("The view info should be a string for {}, but it is: {}".format(base_name, view_info)) 

774 

775 if len(differentiable_output_vars) == 0: 

776 # no output is differentiable (.indices() for SparseTensors for example) 

777 return 'as_view({}, {}, /* is_differentiable */ false)'.format(view_info, call) 

778 elif len(differentiable_output_vars) == 1: 

779 # Single differentiable output (Tensor or Tensor[]) 

780 return_info = differentiable_outputs[0] 

781 # We only support simple Tensor or a TensorList for functions that return views 

782 if not return_info['dynamic_type'] in ['Tensor', 'TensorList']: 

783 raise RuntimeError("{} that return differentiable views can only return Tensor or Tensor[]".format(base_name)) 

784 # Only allow rebasing of the history if we return a single Tensor 

785 allow_rebase_history = 'true' 

786 if return_info['dynamic_type'] == 'TensorList': 

787 allow_rebase_history = 'false' 

788 wrapped_call = ("as_view(/* base */{}, /* output */ {}, /* is_differentiable */ true, " 

789 "/* allow_rebase_history */ {})").format(view_info, call, allow_rebase_history) 

790 return wrapped_call 

791 else: 

792 # This could be supported but we don't need it at the moment, so keeping things simple. 

793 raise RuntimeError("Function that return multiple differentiable output " 

794 "when at least one of them is view is not supported.") 

795 else: 

796 return 'std::move({})'.format(call)

821 def emit_call(env): 

822 combined = nested_dict(env, declaration) 

823 if strategy == 'use_derived': 

824 # We only care about adding `at::AutoNonVariableTypeMode` guard for non-variable dispatch 

825 # (which corresponds to 'use_derived' strategy). The purpose of this guard is to make sure 

826 # the baseType operations still dispatch to non-Variable type, even if the arguments passed 

827 # in are now Variables. 

828 # See NOTE [ Treating Variables as non-Variables in type dispatch ] for details. 

829 if 'namespace' in declaration['method_of']: 

830 base_type_call = CALL_DISPATCH_VIA_NAMESPACE.substitute(combined) 

831 else: 

832 unpacked_method_args = combined['unpacked_args'][1:] 

833 base_type_call = CALL_DISPATCH_VIA_METHOD.substitute( 

834 combined, unpacked_method_args=unpacked_method_args) 

835 if not modifies_arguments and not returns_void: 

836 rhs_value = wrap_output('tmp') 

837 call = DISPATCH_TO_NON_VAR_TYPE_WITH_RETURN_VALUES.substitute( 

838 base_type_call=base_type_call, 

839 return_values=tie_return_values(), 

840 rhs_value=rhs_value) 

841 else: 

842 call = DISPATCH_TO_NON_VAR_TYPE_WITHOUT_RETURN_VALUES.substitute( 

843 base_type_call=base_type_call) 

844 else: 

845 call = CALL_DEFAULT.substitute(declaration) 

846 if not modifies_arguments and not returns_void: 

847 call = '{} = {}'.format(tie_return_values(), call) 

848 call = call + ';' 

849 call = enforce_same_tensorimpl_and_storage(env, call) 

850 return call

984def dispatch_strategy(declaration): 

985 """How are we going to call the underlying implementation of a 

986 declaration? There are two strategies: 

987 

988 - use_derived: we want to call the implementation on CPUDoubleType 

989 (or a similar, derived Type instance). Because these derived 

990 instances deal in Tensors, not Variables (it's a completely different 

991 object, so it doesn't dispatch back to VariableType), code on 

992 this dispatch path needs to wrap/unwrap tensors. If the 

993 derived implementation takes and returns tensors, the 

994 implementation is usually differentiable (although we also use 

995 the derived dispatch path for non-differentiable functions 

996 that we still want to dispatch on the derived Type instance; 

997 e.g., size()) 

998 

999 - use_type: we want to call the implementation on Type, because 

1000 it is implemented concretely, and the functions it invokes will 

1001 get dispatched back to VariableType (which will ensure that they 

1002 are differentiable.) 

1003 """ 

1004 if (declaration['abstract'] or declaration['requires_tensor'] or 

1005 declaration['derivative'] is not None): 

1006 # If the function is abstract (not implemented on at::Type), we must 

1007 # call the implementation on the derived type with unpacked tensors. 

1008 

1009 # If the function has a derivative specified and is concrete, we could 

1010 # call either implementation. We prefer the calling the derived 

1011 # type's implementation with unpacked tensors because it is more 

1012 # performant in some cases: any internal calls to other ATen functions 

1013 # won't have the history tracked. 

1014 

1015 # If the function has a type dispatched argument (i.e. is a factory), 

1016 # we prefer calling the derived type's implementation both because it is 

1017 # more performant and to ensure factory functions return tensors with _version 

1018 # of 0 (probably not strictly necessary, but nice to have to keeps versions simple 

1019 # to understand. 

1020 return 'use_derived' 

1021 else: 

1022 # If the function is concrete (we don't have to override it) and we 

1023 # didn't declare it in derivatives.yaml, we'll assume that it is 

1024 # actually implemented out of differentiable functions. (This 

1025 # assumption might not hold, but then you'll see gradcheck fail.) 

1026 return 'use_type'
