54def get_signature(fn, rcb, loc, is_method): 

55 # Python 3.5 adds support for the nice annotation syntax, so try that first. 

56 signature = None 

57 if PY35: 

58 signature = try_real_annotations(fn) 

59 if signature is not None and is_method: 

60 # If this is a method, then the signaure will include a type for 

61 # `self`, but type comments do not contain a `self`. So strip it 

62 # away here so everything is consistent (`inspect.ismethod` does 

63 # not work here since `fn` is unbound at this point) 

64 param_types, return_type = signature 

65 param_types = param_types[1:] 

66 signature = (param_types, return_type) 

67 

68 if signature is None: 

69 type_line, source = None, None 

70 try: 

71 source = dedent(''.join(get_source_lines_and_file(fn)[0])) 

72 type_line = get_type_line(source) 

73 except TypeError: 

74 pass 

75 # This might happen both because we failed to get the source of fn, or 

76 # because it didn't have any annotations. 

77 if type_line is not None: 

78 signature = parse_type_line(type_line, rcb, loc) 

79 

80 return signature

89def is_vararg(the_callable): 

90 if not is_function_or_method(the_callable) and hasattr(the_callable, '__call__'): # noqa: B004 

91 # If `the_callable` is a class, de-sugar the call so we can still get 

92 # the signature 

93 the_callable = the_callable.__call__ 

94 

95 if is_function_or_method(the_callable): 

96 if PY2:

98 # `inspect.getfullargspec` is not available in Python 2 but 

99 # `inspect.getargspec` is deprecated in Python 3, so we have to 

100 # switch over them 

101 return inspect.getargspec(the_callable).varargs is not None 

102 else: 

103 return inspect.getfullargspec(the_callable).varargs is not None 

104 else:

108def get_param_names(fn, n_args): 

109 if not is_function_or_method(fn) and hasattr(fn, '__call__') and is_function_or_method(fn.__call__): # noqa: B004 

110 # De-sugar calls to classes 

111 fn = fn.__call__ 

112 

113 if is_function_or_method(fn): 

114 if PY2: 

115 # see [inspect args] 

116 return inspect.getargspec(fn).args 

117 else:

119 else:

121 # method, so use a default param name list) 

122 return [str(i) for i in range(n_args)]

142def parse_type_line(type_line, rcb, loc): 

143 """Parses a type annotation specified as a comment. 

144 

145 Example inputs: 

146 # type: (Tensor, torch.Tensor) -> Tuple[Tensor] 

147 # type: (Tensor, Tuple[Tensor, Tensor]) -> Tensor 

148 """ 

149 arg_ann_str, ret_ann_str = split_type_line(type_line) 

150 

151 try: 

152 arg_ann = eval(arg_ann_str, {}, EvalEnv(rcb)) # noqa: P204 

153 except (NameError, SyntaxError) as e: 

154 raise RuntimeError("Failed to parse the argument list of a type annotation: {}".format(str(e))) 

155 

156 if not isinstance(arg_ann, tuple): 

157 arg_ann = (arg_ann,) 

158 

159 try: 

160 ret_ann = eval(ret_ann_str, {}, EvalEnv(rcb)) # noqa: P204 

161 except (NameError, SyntaxError) as e: 

162 raise RuntimeError("Failed to parse the return type of a type annotation: {}".format(str(e))) 

163 

164 resolver = (rcb, loc) 

165 arg_types = [ann_to_type(ann, resolver) for ann in arg_ann] 

166 return arg_types, ann_to_type(ret_ann, resolver)

169def get_type_line(source): 

170 """Tries to find the line containing a comment with the type annotation.""" 

171 type_comment = '# type:' 

172 

173 lines = source.split('\n') 

174 lines = [(line_num, line) for line_num, line in enumerate(lines)] 

175 type_lines = list(filter(lambda line: type_comment in line[1], lines)) 

176 lines_with_type = list(filter(lambda line: 'type' in line[1], lines)) 

177 

178 if len(type_lines) == 0: 

179 type_pattern = re.compile('#[\t ]*type[\t ]*:') 

180 wrong_type_lines = list(filter(lambda line: type_pattern.search(line[1]), lines)) 

181 if len(wrong_type_lines) > 0: 

182 raise RuntimeError("The annotation prefix in line " + str(wrong_type_lines[0][0]) 

183 + " is probably invalid.\nIt must be '# type:'" 

184 + "\nSee PEP 484 (https://www.python.org/dev/peps/pep-0484/#suggested-syntax-for-python-2-7-and-straddling-code)" # noqa 

185 + "\nfor examples") 

186 return None 

187 elif len(type_lines) == 1: 

188 # Only 1 type line, quit now 

189 return type_lines[0][1].strip() 

190 

191 # Parse split up argument types according to PEP 484 

192 # https://www.python.org/dev/peps/pep-0484/#suggested-syntax-for-python-2-7-and-straddling-code 

193 return_line = None 

194 parameter_type_lines = [] 

195 for line_num, line in type_lines: 

196 if '# type: (...) -> ' in line: 

197 return_line = (line_num, line) 

198 break 

199 elif type_comment in line: 

200 parameter_type_lines.append(line) 

201 if return_line is None: 

202 raise RuntimeError("Return type line '# type: (...) -> ...' not found on multiline " 

203 "type annotation\n(See PEP 484 https://www.python.org/dev/peps/pep-0484/#suggested-syntax-for-python-2-7-and-straddling-code)") # noqa 

204 

205 def get_parameter_type(line): 

206 item_type = line[line.find(type_comment) + len(type_comment):] 

207 return item_type.strip() 

208 

209 types = map(get_parameter_type, parameter_type_lines) 

210 parameter_types = ", ".join(types) 

211 

212 return return_line[1].replace("...", parameter_types)

215def split_type_line(type_line): 

216 """Splits the comment with the type annotation into parts for argument and return types. 

217 

218 For example, for an input of: 

219 # type: (Tensor, torch.Tensor) -> Tuple[Tensor, Tensor] 

220 

221 This function will return: 

222 ("(Tensor, torch.Tensor)", "Tuple[Tensor, Tensor]") 

223 

224 """ 

225 start_offset = len('# type:') 

226 try: 

227 arrow_pos = type_line.index('->') 

228 except ValueError: 

229 raise RuntimeError("Syntax error in type annotation (cound't find `->`)") 

230 return type_line[start_offset:arrow_pos].strip(), type_line[arrow_pos + 2:].strip()

233def try_real_annotations(fn): 

234 """Tries to use the Py3.5+ annotation syntax to get the type.""" 

235 try: 

236 sig = inspect.signature(fn) 

237 except ValueError: 

238 return None 

239 

240 all_annots = [sig.return_annotation] + [p.annotation for p in sig.parameters.values()] 

241 if all(ann is sig.empty for ann in all_annots): 

242 return None 

243 

244 def as_ann(ann): 

245 # sig.empty is really annoying so convert it to None 

246 return ann if ann is not sig.empty else None 

247 

248 arg_types = [ann_to_type(as_ann(p.annotation)) 

249 for p in sig.parameters.values()] 

250 return_type = ann_to_type(as_ann(sig.return_annotation)) 

251 return arg_types, return_type

254def ann_to_type(ann, resolver=None): 

255 # resolver should be a Tuple[Callable, SourceRange] where the Callable 

256 # is a resolutionCallback 

257 if ann is None: 

258 return TensorType.get() 

259 elif ann is torch.Tensor: 

260 return TensorType.get() 

261 elif is_tuple(ann): 

262 return TupleType([ann_to_type(a) for a in ann.__args__]) 

263 elif is_list(ann): 

264 return ListType(ann_to_type(ann.__args__[0])) 

265 elif is_dict(ann): 

266 key = ann_to_type(ann.__args__[0]) 

267 value = ann_to_type(ann.__args__[1]) 

268 return DictType(key, value) 

269 elif is_optional(ann): 

270 if issubclass(ann.__args__[1], type(None)): 

271 return OptionalType(ann_to_type(ann.__args__[0])) 

272 else: 

273 return OptionalType(ann_to_type(ann.__args__[1])) 

274 elif ann is float: 

275 return FloatType.get() 

276 elif ann is int: 

277 return IntType.get() 

278 elif ann is str: 

279 return StringType.get() 

280 elif ann is bool: 

281 return BoolType.get() 

282 elif ann is Any: 

283 return AnyType.get() 

284 elif ann is type(None): 

285 return NoneType.get() 

286 elif hasattr(ann, "__torch_script_class__"): 

287 return ClassType(_qualified_name(ann)) 

288 elif hasattr(ann, "__torch_script_interface__"): 

289 return InterfaceType(_qualified_name(ann)) 

290 elif ann is torch.device: 

291 return DeviceObjType.get() 

292 elif resolver is not None: 

293 # Maybe resolve a NamedTuple to a Tuple Type 

294 rcb, loc = resolver 

295 the_type = torch._C._resolve_type(ann.__name__, loc, rcb) 

296 if the_type is not None: 

297 return the_type 

298 raise ValueError("Unknown type annotation: '{}'".format(ann))
