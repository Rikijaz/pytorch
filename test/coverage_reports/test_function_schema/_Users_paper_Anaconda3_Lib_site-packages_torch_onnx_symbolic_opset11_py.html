340# Generate paddings in ONNX order based on pad in pytorch. 

341# Arguments: 

342# dim: the dimension of the tensor. 

343# pad: the paddings in pytorch. 

344# The order is dim_n_begin, dim_n_end, dim_n-1_begin, dim_n-1_end, ..., dim_m_begin, dim_m_end, 

345# where m is in range [0, n]. 

346def _prepare_onnx_paddings(g, dim, pad): 

347 # The desired order of paddings is 

348 # dim_0_begin, dim_1_begin, ... , dim_0_end, ..., dim_n_end. 

349 # n is the dimension of input. 

350 # Assume zero-dimensions in the beginning, pad the "pad" sequence with zeros in the beginning 

351 pad_len = torch.onnx.symbolic_opset9.size(g, pad, g.op("Constant", value_t=torch.tensor([0]))) 

352 # Set extension = [0] * (dim * 2 - len(pad)) 

353 extension = g.op("Sub", g.op("Mul", g.op("Constant", value_t=torch.tensor(dim, dtype=torch.int64)), 

354 g.op("Constant", value_t=torch.tensor(2, dtype=torch.int64))), pad_len) 

355 # Concat pad with extension: paddings = [dim_n_begin, dim_n_end, dim_n-1_begin, dim_n-1_end, 0, 0, ... ] 

356 # Currently ONNX only supports int64 type for Pad 

357 pad = g.op("Cast", pad, to_i=sym_help.cast_pytorch_to_onnx['Long']) 

358 paddings = g.op("Concat", pad, g.op("ConstantOfShape", extension, value_t=torch.tensor([0], dtype=torch.int64)), axis_i=0) 

359 # Reshape and reverse order and collate first beginnings and then ends 

360 # paddings = [[..., 0, dim_n-1_begin, dim_n_begin], 

361 # [..., 0, dim_n-1_end, dim_n_end]] 

362 # Reshape back to 1-D paddings = [..., 0, dim_n - 1_begin, dim_n_begin, ..., 0, dim_n - 1_end, dim_n_end] 

363 paddings = g.op("Reshape", paddings, g.op("Constant", value_t=torch.tensor([-1, 2]))) 

364 paddings = g.op("Transpose", torch.onnx.symbolic_opset10.flip(g, paddings, [0]), perm_i=[1, 0]) 

365 paddings = g.op("Reshape", paddings, g.op("Constant", value_t=torch.tensor([-1]))) 

366 padding_c = g.op("Cast", paddings, to_i=sym_help.cast_pytorch_to_onnx['Long']) 

367 return padding_c

471 return g.op("ATen", self, index, value, dim_i=dim_value, operator_s="index_fill") 

472 expanded_index_shape, expanded_index = sym_help._index_fill_reshape_helper(g, self, dim, index) 

473 value = sym_help._maybe_get_scalar(value) 

474 value = sym_help._if_scalar_type_as(g, value, self) 

475 expanded_value = expand(g, value, expanded_index_shape, None) 

476 return scatter(g, self, dim, expanded_index, expanded_value)

487def __rshift_(g, self, other): 

488 # make sure to cast other to self's type 

489 # (when self is long, make sure that other is not float) 

490 if other.type().scalarType() != self.type().scalarType(): 

491 other = g.op("Cast", other, to_i=sym_help.cast_pytorch_to_onnx[self.type().scalarType()]) 

492 

493 if self.type().scalarType() == 'Byte': 

494 return g.op('BitShift', self, other, direction_s="RIGHT") 

495 

496 two = g.op('Constant', value_t=torch.tensor(2, dtype=torch.float32)) 

497 # exponent (same type as self) has to be float or double in onnx::Pow 

498 if not sym_help._is_fp(self): 

499 other = g.op("Cast", other, to_i=sym_help.cast_pytorch_to_onnx['Float']) 

500 two_pow = g.op('Pow', two, other) 

501 

502 rshift = g.op('Div', self, two_pow) 

503 return rshift

525def _get_im2col_indices_along_dim(g, input_d, kernel_size_d, dilation_d, padding_d, stride_d): 

526 # Input is always 4-D (N, C, H, W) 

527 # Calculate indices of sliding blocks along spatial dimension 

528 # Slide kernel over input each dim d: 

529 # each dimension d ranges from 0 to input[d]+2xpadding[d]-dilation[d]x(kernel_size[d]-1) 

530 # with steps = stride 

531 

532 blocks_d = g.op("Add", input_d, g.op("Constant", value_t=torch.tensor(padding_d * 2))) 

533 blocks_d = g.op("Sub", blocks_d, g.op("Constant", value_t=torch.tensor(dilation_d * (kernel_size_d - 1)))) 

534 

535 # Stride kernel over input and find starting indices along dim d 

536 blocks_d_indices = g.op("Range", g.op("Constant", value_t=torch.tensor(0)), 

537 blocks_d, g.op("Constant", value_t=torch.tensor(stride_d))) 

538 

539 # Apply dilation on kernel and find its indices along dim d 

540 kernel_grid = numpy.arange(0, kernel_size_d * dilation_d, dilation_d) 

541 kernel_grid = g.op("Constant", value_t=torch.tensor([kernel_grid])) 

542 

543 # Broadcast and add kernel staring positions (indices) with 

544 # kernel_grid along dim d, to get block indices along dim d 

545 blocks_d_indices = g.op('Unsqueeze', blocks_d_indices, axes_i=[0]) # Reshape to [1, -1] 

546 kernel_mask = g.op('Reshape', kernel_grid, g.op('Constant', value_t=torch.tensor([-1, 1]))) 

547 block_mask = g.op("Add", blocks_d_indices, kernel_mask) 

548 

549 return block_mask

552def _get_im2col_padded_input(g, input, padding_h, padding_w): 

553 # Input is always 4-D tensor (N, C, H, W) 

554 # Padding tensor has the following format: (padding_h, padding_w) 

555 # Reshape the padding to follow ONNX format: (dim1_begin, dim2_begin,...,dim1_end, dim2_end,...) 

556 pad = g.op("Constant", value_t=torch.LongTensor([0, 0, padding_h, padding_w] * 2)) 

557 return g.op("Pad", input, pad)

560def _get_im2col_output_shape(g, input, kernel_h, kernel_w): 

561 batch_dim = size(g, input, g.op("Constant", value_t=torch.tensor(0))) 

562 channel_dim = size(g, input, g.op("Constant", value_t=torch.tensor(1))) 

563 channel_unfolded = g.op("Mul", channel_dim, 

564 g.op("Constant", value_t=torch.tensor(kernel_h * kernel_w))) 

565 

566 return g.op("Concat", 

567 g.op("Unsqueeze", batch_dim, axes_i=[0]), 

568 g.op("Unsqueeze", channel_unfolded, axes_i=[0]), 

569 g.op("Constant", value_t=torch.tensor([-1])), axis_i=0)

572@parse_args('v', 'is', 'is', 'is', 'is') 

573def im2col(g, input, kernel_size, dilation, padding, stride): 

574 # Input is always 4-D tensor (N, C, H, W) 

575 # All other args are int[2] 

576 

577 input_h = size(g, input, g.op("Constant", value_t=torch.tensor(2))) 

578 input_w = size(g, input, g.op("Constant", value_t=torch.tensor(3))) 

579 

580 stride_h, stride_w = stride[0], stride[1] 

581 padding_h, padding_w = padding[0], padding[1] 

582 dilation_h, dilation_w = dilation[0], dilation[1] 

583 kernel_h, kernel_w = kernel_size[0], kernel_size[1] 

584 

585 blocks_row_indices = _get_im2col_indices_along_dim(g, input_h, kernel_h, dilation_h, padding_h, stride_h) 

586 blocks_col_indices = _get_im2col_indices_along_dim(g, input_w, kernel_w, dilation_w, padding_w, stride_w) 

587 

588 output_shape = _get_im2col_output_shape(g, input, kernel_h, kernel_w) 

589 padded_input = _get_im2col_padded_input(g, input, padding_h, padding_w) 

590 

591 # For a 4D matrix of size (1, 1, 3, 3) as below with kernel_size=2, stride=1, and dilation=1 

592 # [[[[1., 2., 3.,], 

593 # [4., 5., 6.,], 

594 # [7., 8., 9.,]]]] 

595 # First gather indices along rows (dim=2) with blocks_row_indices = [[0,1], [1,2]] to get: 

596 # [[[[[1., 2., 3.], 

597 # [4., 5., 6.]], 

598 # [[4., 5., 6.], 

599 # [7., 8., 9.]]]]] 

600 # And then gather along cols (dim=4) with blocks_row_indices = [[0,1], [1,2]] to get: 

601 # [[[[[[1., 2.], 

602 # [4., 5.]], 

603 # [[2., 3.], 

604 # [5., 6]]], 

605 # [[[4., 5.], 

606 # [7., 8.]], 

607 # [[5., 6.], 

608 # [8., 9.]]]]]] 

609 # Transpose dims 3 (depth) and 4 (rows), and then reshape to output shape (1, 1, 4, 4) to get: 

610 # [[[1., 2., 4., 5.], 

611 # [2., 3., 5., 6.], 

612 # [4., 5., 7., 8.], 

613 # [5., 6., 8., 9.]]] 

614 output = g.op("Gather", padded_input, blocks_row_indices, axis_i=2) 

615 output = g.op("Gather", output, blocks_col_indices, axis_i=4) 

616 output = g.op("Transpose", output, perm_i=[0, 1, 2, 4, 3, 5]) 

617 return g.op("Reshape", output, output_shape)

620@parse_args('v', 'i', 'i') 

621def flatten(g, input, start_dim, end_dim): 

622 dim = input.type().dim() 

623 # use ONNX's Flatten operator for cases where the output shape is 2D 

624 if start_dim == 1: 

625 if (end_dim == -1 or (end_dim is not None and end_dim == dim - 1)): 

626 return g.op("Flatten", input, axis_i=start_dim) 

627 elif start_dim == 0: 

628 if (end_dim == -2 or (end_dim is not None and end_dim == dim - 2)): 

629 return g.op("Flatten", input, axis_i=end_dim + 1) 

630 # use Reshape for cases where the output shape is not 2D 

631 if not input.isCompleteTensor(): 

632 return _unimplemented("flatten", 

633 "input size not accessible " 

634 "(consider using reshape op instead of flatten op to export to ONNX)") 

635 # if end_dim is negative add dim 

636 if end_dim < 0 : 

637 end_dim = dim + end_dim 

638 input_dims = input.type().sizes() 

639 output_dims = [] 

640 for i in range(0, dim): 

641 if start_dim < i and end_dim >= i: 

642 output_dims[start_dim] = output_dims[start_dim] * input_dims[i] 

643 else: 

644 output_dims.append(input_dims[i]) 

645 shape = g.op("Constant", value_t=torch.LongTensor(output_dims)) 

646 from torch.onnx.symbolic_opset9 import _reshape_from_tensor 

647 p = _reshape_from_tensor(g, input, shape) 

648 return p
