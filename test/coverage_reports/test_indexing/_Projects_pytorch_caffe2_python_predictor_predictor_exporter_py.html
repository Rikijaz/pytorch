135def prepare_prediction_net(filename, db_type, device_option=None): 

136 ''' 

137 Helper function which loads all required blobs from the db 

138 and returns prediction net ready to be used 

139 ''' 

140 metanet_def = load_from_db(filename, db_type, device_option) 

141 

142 global_init_net = utils.GetNet( 

143 metanet_def, predictor_constants.GLOBAL_INIT_NET_TYPE) 

144 workspace.RunNetOnce(global_init_net) 

145 

146 predict_init_net = utils.GetNet( 

147 metanet_def, predictor_constants.PREDICT_INIT_NET_TYPE) 

148 workspace.RunNetOnce(predict_init_net) 

149 

150 predict_net = core.Net( 

151 utils.GetNet(metanet_def, predictor_constants.PREDICT_NET_TYPE)) 

152 workspace.CreateNet(predict_net) 

153 

154 return predict_net

157def _global_init_net(predictor_export_meta, db_type): 

158 net = core.Net("global-init") 

159 # manifold_db does not need DBReader 

160 if db_type != "manifold_db": 

161 net.Load( 

162 [predictor_constants.PREDICTOR_DBREADER], 

163 predictor_export_meta.parameters) 

164 net.Proto().external_input.extend([predictor_constants.PREDICTOR_DBREADER]) 

165 net.Proto().external_output.extend(predictor_export_meta.parameters) 

166 

167 if predictor_export_meta.global_init_net: 

168 net.AppendNet(predictor_export_meta.global_init_net) 

169 

170 # Add the model_id in the predict_net to the global_init_net 

171 utils.AddModelIdArg(predictor_export_meta, net.Proto()) 

172 return net.Proto()

176 """ 

177 """

230def load_from_db(filename, db_type, device_option=None, *args, **kwargs): 

231 # global_init_net in meta_net_def will load parameters from 

232 # predictor_constants.PREDICTOR_DBREADER 

233 create_db = core.CreateOperator( 

234 'CreateDB', [], 

235 [core.BlobReference(predictor_constants.PREDICTOR_DBREADER)], 

236 db=filename, db_type=db_type) 

237 assert workspace.RunOperatorOnce(create_db), ( 

238 'Failed to create db {}'.format(filename)) 

239 

240 # predictor_constants.META_NET_DEF is always stored before the parameters 

241 load_meta_net_def = core.CreateOperator( 

242 'Load', 

243 [core.BlobReference(predictor_constants.PREDICTOR_DBREADER)], 

244 [core.BlobReference(predictor_constants.META_NET_DEF)]) 

245 assert workspace.RunOperatorOnce(load_meta_net_def) 

246 

247 blob = workspace.FetchBlob(predictor_constants.META_NET_DEF) 

248 meta_net_def = serde.deserialize_protobuf_struct( 

249 blob if isinstance(blob, bytes) 

250 else str(blob).encode('utf-8'), 

251 metanet_pb2.MetaNetDef) 

252 

253 if device_option is None: 

254 device_option = scope.CurrentDeviceScope() 

255 

256 if device_option is not None: 

257 # Set the device options of all loaded blobs 

258 for kv in meta_net_def.nets: 

259 net = kv.value 

260 for op in net.op: 

261 op.device_option.CopyFrom(device_option) 

262 

263 return meta_net_def
