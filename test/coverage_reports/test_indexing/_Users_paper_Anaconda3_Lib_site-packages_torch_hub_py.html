138def _get_cache_or_reload(github, force_reload, verbose=True): 

139 # Parse github repo information 

140 repo_owner, repo_name, branch = _parse_repo_info(github) 

141 # Github allows branch name with slash '/', 

142 # this causes confusion with path on both Linux and Windows. 

143 # Backslash is not allowed in Github branch name so no need to 

144 # to worry about it. 

145 normalized_br = branch.replace('/', '_') 

146 # Github renames folder repo-v1.x.x to repo-1.x.x 

147 # We don't know the repo name before downloading the zip file 

148 # and inspect name from it. 

149 # To check if cached repo exists, we need to normalize folder names. 

150 repo_dir = os.path.join(hub_dir, '_'.join([repo_owner, repo_name, normalized_br])) 

151 

152 use_cache = (not force_reload) and os.path.exists(repo_dir) 

153 

154 if use_cache: 

155 if verbose: 

156 sys.stderr.write('Using cache found in {}\n'.format(repo_dir)) 

157 else: 

158 cached_file = os.path.join(hub_dir, normalized_br + '.zip') 

159 _remove_if_exists(cached_file) 

160 

161 url = _git_archive_link(repo_owner, repo_name, branch) 

162 sys.stderr.write('Downloading: \"{}\" to {}\n'.format(url, cached_file)) 

163 download_url_to_file(url, cached_file, progress=False) 

164 

165 with zipfile.ZipFile(cached_file) as cached_zipfile: 

166 extraced_repo_name = cached_zipfile.infolist()[0].filename 

167 extracted_repo = os.path.join(hub_dir, extraced_repo_name) 

168 _remove_if_exists(extracted_repo) 

169 # Unzip the code and rename the base folder 

170 cached_zipfile.extractall(hub_dir) 

171 

172 _remove_if_exists(cached_file) 

173 _remove_if_exists(repo_dir) 

174 shutil.move(extracted_repo, repo_dir) # rename the repo 

175 

176 return repo_dir

179def _check_module_exists(name): 

180 if sys.version_info >= (3, 4): 

181 import importlib.util 

182 return importlib.util.find_spec(name) is not None 

183 elif sys.version_info >= (3, 3): 

184 # Special case for python3.3 

185 import importlib.find_loader 

186 return importlib.find_loader(name) is not None 

187 else: 

188 # NB: Python2.7 imp.find_module() doesn't respect PEP 302, 

189 # it cannot find a package installed as .egg(zip) file. 

190 # Here we use workaround from: 

191 # https://stackoverflow.com/questions/28962344/imp-find-module-which-supports-zipped-eggs?lq=1 

192 # Also imp doesn't handle hierarchical module names (names contains dots). 

193 try: 

194 # 1. Try imp.find_module(), which searches sys.path, but does 

195 # not respect PEP 302 import hooks. 

196 import imp 

197 result = imp.find_module(name) 

198 if result: 

199 return True 

200 except ImportError: 

201 pass 

202 path = sys.path 

203 for item in path: 

204 # 2. Scan path for import hooks. sys.path_importer_cache maps 

205 # path items to optional "importer" objects, that implement 

206 # find_module() etc. Note that path must be a subset of 

207 # sys.path for this to work. 

208 importer = sys.path_importer_cache.get(item) 

209 if importer: 

210 try: 

211 result = importer.find_module(name, [item]) 

212 if result: 

213 return True 

214 except ImportError: 

215 pass 

216 return False

376def download_url_to_file(url, dst, hash_prefix=None, progress=True): 

377 r"""Download object at the given URL to a local path. 

378 

379 Args: 

380 url (string): URL of the object to download 

381 dst (string): Full path where object will be saved, e.g. `/tmp/temporary_file` 

382 hash_prefix (string, optional): If not None, the SHA256 downloaded file should start with `hash_prefix`. 

383 Default: None 

384 progress (bool, optional): whether or not to display a progress bar to stderr 

385 Default: True 

386 

387 Example: 

388 >>> torch.hub.download_url_to_file('https://s3.amazonaws.com/pytorch/models/resnet18-5c106cde.pth', '/tmp/temporary_file') 

389 

390 """ 

391 file_size = None 

392 # We use a different API for python2 since urllib(2) doesn't recognize the CA 

393 # certificates in older Python 

394 u = urlopen(url) 

395 meta = u.info() 

396 if hasattr(meta, 'getheaders'): 

397 content_length = meta.getheaders("Content-Length") 

398 else: 

399 content_length = meta.get_all("Content-Length") 

400 if content_length is not None and len(content_length) > 0: 

401 file_size = int(content_length[0]) 

402 

403 # We deliberately save it in a temp file and move it after 

404 # download is complete. This prevents a local working checkpoint 

405 # being overridden by a broken download. 

406 dst = os.path.expanduser(dst) 

407 dst_dir = os.path.dirname(dst) 

408 f = tempfile.NamedTemporaryFile(delete=False, dir=dst_dir) 

409 

410 try: 

411 if hash_prefix is not None: 

412 sha256 = hashlib.sha256() 

413 with tqdm(total=file_size, disable=not progress, 

414 unit='B', unit_scale=True, unit_divisor=1024) as pbar: 

415 while True: 

416 buffer = u.read(8192) 

417 if len(buffer) == 0: 

418 break 

419 f.write(buffer) 

420 if hash_prefix is not None: 

421 sha256.update(buffer) 

422 pbar.update(len(buffer)) 

423 

424 f.close() 

425 if hash_prefix is not None: 

426 digest = sha256.hexdigest() 

427 if digest[:len(hash_prefix)] != hash_prefix: 

428 raise RuntimeError('invalid hash value (expected "{}", got "{}")' 

429 .format(hash_prefix, digest)) 

430 shutil.move(f.name, dst) 

431 finally: 

432 f.close() 

433 if os.path.exists(f.name): 

434 os.remove(f.name)

442def load_state_dict_from_url(url, model_dir=None, map_location=None, progress=True, check_hash=False): 

443 r"""Loads the Torch serialized object at the given URL. 

444 

445 If downloaded file is a zip file, it will be automatically 

446 decompressed. 

447 

448 If the object is already present in `model_dir`, it's deserialized and 

449 returned. 

450 The default value of `model_dir` is ``$TORCH_HOME/checkpoints`` where 

451 environment variable ``$TORCH_HOME`` defaults to ``$XDG_CACHE_HOME/torch``. 

452 ``$XDG_CACHE_HOME`` follows the X Design Group specification of the Linux 

453 filesytem layout, with a default value ``~/.cache`` if not set. 

454 

455 Args: 

456 url (string): URL of the object to download 

457 model_dir (string, optional): directory in which to save the object 

458 map_location (optional): a function or a dict specifying how to remap storage locations (see torch.load) 

459 progress (bool, optional): whether or not to display a progress bar to stderr. 

460 Default: True 

461 check_hash(bool, optional): If True, the filename part of the URL should follow the naming convention 

462 ``filename-<sha256>.ext`` where ``<sha256>`` is the first eight or more 

463 digits of the SHA256 hash of the contents of the file. The hash is used to 

464 ensure unique names and to verify the contents of the file. 

465 Default: False 

466 

467 Example: 

468 >>> state_dict = torch.hub.load_state_dict_from_url('https://s3.amazonaws.com/pytorch/models/resnet18-5c106cde.pth') 

469 

470 """ 

471 # Issue warning to move data if old env is set 

472 if os.getenv('TORCH_MODEL_ZOO'): 

473 warnings.warn('TORCH_MODEL_ZOO is deprecated, please use env TORCH_HOME instead') 

474 

475 if model_dir is None: 

476 torch_home = _get_torch_home() 

477 model_dir = os.path.join(torch_home, 'checkpoints') 

478 

479 try: 

480 os.makedirs(model_dir) 

481 except OSError as e: 

482 if e.errno == errno.EEXIST: 

483 # Directory already exists, ignore. 

484 pass 

485 else: 

486 # Unexpected OSError, re-raise. 

487 raise 

488 

489 parts = urlparse(url) 

490 filename = os.path.basename(parts.path) 

491 cached_file = os.path.join(model_dir, filename) 

492 if not os.path.exists(cached_file): 

493 sys.stderr.write('Downloading: "{}" to {}\n'.format(url, cached_file)) 

494 hash_prefix = HASH_REGEX.search(filename).group(1) if check_hash else None 

495 download_url_to_file(url, cached_file, hash_prefix, progress=progress) 

496 

497 # Note: extractall() defaults to overwrite file if exists. No need to clean up beforehand. 

498 # We deliberately don't handle tarfile here since our legacy serialization format was in tar. 

499 # E.g. resnet18-5c106cde.pth which is widely used. 

500 if zipfile.is_zipfile(cached_file): 

501 with zipfile.ZipFile(cached_file) as cached_zipfile: 

502 members = cached_zipfile.infolist() 

503 if len(members) != 1: 

504 raise RuntimeError('Only one file(not dir) is allowed in the zipfile') 

505 cached_zipfile.extractall(model_dir) 

506 extraced_name = members[0].filename 

507 cached_file = os.path.join(model_dir, extraced_name) 

508 

509 return torch.load(cached_file, map_location=map_location)
