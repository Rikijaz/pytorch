6def _cubic_interpolate(x1, f1, g1, x2, f2, g2, bounds=None): 

7 # ported from https://github.com/torch/optim/blob/master/polyinterp.lua 

8 # Compute bounds of interpolation area 

9 if bounds is not None: 

10 xmin_bound, xmax_bound = bounds 

11 else: 

12 xmin_bound, xmax_bound = (x1, x2) if x1 <= x2 else (x2, x1) 

13 

14 # Code for most common case: cubic interpolation of 2 points 

15 # w/ function and derivative values for both 

16 # Solution in this case (where x2 is the farthest point): 

17 # d1 = g1 + g2 - 3*(f1-f2)/(x1-x2); 

18 # d2 = sqrt(d1^2 - g1*g2); 

19 # min_pos = x2 - (x2 - x1)*((g2 + d2 - d1)/(g2 - g1 + 2*d2)); 

20 # t_new = min(max(min_pos,xmin_bound),xmax_bound); 

21 d1 = g1 + g2 - 3 * (f1 - f2) / (x1 - x2) 

22 d2_square = d1**2 - g1 * g2 

23 if d2_square >= 0: 

24 d2 = d2_square.sqrt() 

25 if x1 <= x2: 

26 min_pos = x2 - (x2 - x1) * ((g2 + d2 - d1) / (g2 - g1 + 2 * d2)) 

27 else: 

28 min_pos = x1 - (x1 - x2) * ((g1 + d2 - d1) / (g1 - g2 + 2 * d2)) 

29 return min(max(min_pos, xmin_bound), xmax_bound) 

30 else: 

31 return (xmin_bound + xmax_bound) / 2.

34def _strong_wolfe(obj_func, 

35 x, 

36 t, 

37 d, 

38 f, 

39 g, 

40 gtd, 

41 c1=1e-4, 

42 c2=0.9, 

43 tolerance_change=1e-9, 

44 max_ls=25): 

45 # ported from https://github.com/torch/optim/blob/master/lswolfe.lua 

46 d_norm = d.abs().max() 

47 g = g.clone(memory_format=torch.contiguous_format) 

48 # evaluate objective and gradient using initial step 

49 f_new, g_new = obj_func(x, t, d) 

50 ls_func_evals = 1 

51 gtd_new = g_new.dot(d) 

52 

53 # bracket an interval containing a point satisfying the Wolfe criteria 

54 t_prev, f_prev, g_prev, gtd_prev = 0, f, g, gtd 

55 done = False 

56 ls_iter = 0 

57 while ls_iter < max_ls: 

58 # check conditions 

59 if f_new > (f + c1 * t * gtd) or (ls_iter > 1 and f_new >= f_prev): 

60 bracket = [t_prev, t] 

61 bracket_f = [f_prev, f_new] 

62 bracket_g = [g_prev, g_new.clone(memory_format=torch.contiguous_format)] 

63 bracket_gtd = [gtd_prev, gtd_new] 

64 break 

65 

66 if abs(gtd_new) <= -c2 * gtd: 

67 bracket = [t] 

68 bracket_f = [f_new] 

69 bracket_g = [g_new] 

70 done = True 

71 break 

72 

73 if gtd_new >= 0: 

74 bracket = [t_prev, t] 

75 bracket_f = [f_prev, f_new] 

76 bracket_g = [g_prev, g_new.clone(memory_format=torch.contiguous_format)] 

77 bracket_gtd = [gtd_prev, gtd_new] 

78 break 

79 

80 # interpolate 

81 min_step = t + 0.01 * (t - t_prev) 

82 max_step = t * 10 

83 tmp = t 

84 t = _cubic_interpolate( 

85 t_prev, 

86 f_prev, 

87 gtd_prev, 

88 t, 

89 f_new, 

90 gtd_new, 

91 bounds=(min_step, max_step)) 

92 

93 # next step 

94 t_prev = tmp 

95 f_prev = f_new 

96 g_prev = g_new.clone(memory_format=torch.contiguous_format) 

97 gtd_prev = gtd_new 

98 f_new, g_new = obj_func(x, t, d) 

99 ls_func_evals += 1 

100 gtd_new = g_new.dot(d) 

101 ls_iter += 1 

102 

103 # reached max number of iterations? 

104 if ls_iter == max_ls: 

105 bracket = [0, t] 

106 bracket_f = [f, f_new] 

107 bracket_g = [g, g_new] 

108 

109 # zoom phase: we now have a point satisfying the criteria, or 

110 # a bracket around it. We refine the bracket until we find the 

111 # exact point satisfying the criteria 

112 insuf_progress = False 

113 # find high and low points in bracket 

114 low_pos, high_pos = (0, 1) if bracket_f[0] <= bracket_f[-1] else (1, 0) 

115 while not done and ls_iter < max_ls: 

116 # compute new trial value 

117 t = _cubic_interpolate(bracket[0], bracket_f[0], bracket_gtd[0], 

118 bracket[1], bracket_f[1], bracket_gtd[1]) 

119 

120 # test that we are making sufficient progress: 

121 # in case `t` is so close to boundary, we mark that we are making 

122 # insufficient progress, and if 

123 # + we have made insufficient progress in the last step, or 

124 # + `t` is at one of the boundary, 

125 # we will move `t` to a position which is `0.1 * len(bracket)` 

126 # away from the nearest boundary point. 

127 eps = 0.1 * (max(bracket) - min(bracket)) 

128 if min(max(bracket) - t, t - min(bracket)) < eps: 

129 # interpolation close to boundary 

130 if insuf_progress or t >= max(bracket) or t <= min(bracket): 

131 # evaluate at 0.1 away from boundary 

132 if abs(t - max(bracket)) < abs(t - min(bracket)): 

133 t = max(bracket) - eps 

134 else: 

135 t = min(bracket) + eps 

136 insuf_progress = False 

137 else: 

138 insuf_progress = True 

139 else: 

140 insuf_progress = False 

141 

142 # Evaluate new point 

143 f_new, g_new = obj_func(x, t, d) 

144 ls_func_evals += 1 

145 gtd_new = g_new.dot(d) 

146 ls_iter += 1 

147 

148 if f_new > (f + c1 * t * gtd) or f_new >= bracket_f[low_pos]: 

149 # Armijo condition not satisfied or not lower than lowest point 

150 bracket[high_pos] = t 

151 bracket_f[high_pos] = f_new 

152 bracket_g[high_pos] = g_new.clone(memory_format=torch.contiguous_format) 

153 bracket_gtd[high_pos] = gtd_new 

154 low_pos, high_pos = (0, 1) if bracket_f[0] <= bracket_f[1] else (1, 0) 

155 else: 

156 if abs(gtd_new) <= -c2 * gtd: 

157 # Wolfe conditions satisfied 

158 done = True 

159 elif gtd_new * (bracket[high_pos] - bracket[low_pos]) >= 0: 

160 # old high becomes new low 

161 bracket[high_pos] = bracket[low_pos] 

162 bracket_f[high_pos] = bracket_f[low_pos] 

163 bracket_g[high_pos] = bracket_g[low_pos] 

164 bracket_gtd[high_pos] = bracket_gtd[low_pos] 

165 

166 # new point becomes new low 

167 bracket[low_pos] = t 

168 bracket_f[low_pos] = f_new 

169 bracket_g[low_pos] = g_new.clone(memory_format=torch.contiguous_format) 

170 bracket_gtd[low_pos] = gtd_new 

171 

172 # line-search bracket is so small 

173 if abs(bracket[1] - bracket[0]) * d_norm < tolerance_change: 

174 break 

175 

176 # return stuff 

177 t = bracket[low_pos] 

178 f_new = bracket_f[low_pos] 

179 g_new = bracket_g[low_pos] 

180 return f_new, g_new, t, ls_func_evals

183class LBFGS(Optimizer): 

184 """Implements L-BFGS algorithm, heavily inspired by `minFunc 

185 <https://www.cs.ubc.ca/~schmidtm/Software/minFunc.html>`. 

186 

187 .. warning:: 

188 This optimizer doesn't support per-parameter options and parameter 

189 groups (there can be only one). 

190 

191 .. warning:: 

192 Right now all parameters have to be on a single device. This will be 

193 improved in the future. 

194 

195 .. note:: 

196 This is a very memory intensive optimizer (it requires additional 

197 ``param_bytes * (history_size + 1)`` bytes). If it doesn't fit in memory 

198 try reducing the history size, or use a different algorithm. 

199 

200 Arguments: 

201 lr (float): learning rate (default: 1) 

202 max_iter (int): maximal number of iterations per optimization step 

203 (default: 20) 

204 max_eval (int): maximal number of function evaluations per optimization 

205 step (default: max_iter * 1.25). 

206 tolerance_grad (float): termination tolerance on first order optimality 

207 (default: 1e-5). 

208 tolerance_change (float): termination tolerance on function 

209 value/parameter changes (default: 1e-9). 

210 history_size (int): update history size (default: 100). 

211 line_search_fn (str): either 'strong_wolfe' or None (default: None). 

212 """

247 def _gather_flat_grad(self): 

248 views = [] 

249 for p in self._params: 

250 if p.grad is None: 

251 view = p.new(p.numel()).zero_() 

252 elif p.grad.is_sparse: 

253 view = p.grad.to_dense().view(-1) 

254 else: 

255 view = p.grad.view(-1) 

256 views.append(view) 

257 return torch.cat(views, 0)

259 def _add_grad(self, step_size, update): 

260 offset = 0 

261 for p in self._params: 

262 numel = p.numel() 

263 # view as to avoid deprecated pointwise semantics 

264 p.data.add_(step_size, update[offset:offset + numel].view_as(p.data)) 

265 offset += numel 

266 assert offset == self._numel()

268 def _clone_param(self): 

269 return [p.clone(memory_format=torch.contiguous_format) for p in self._params]

271 def _set_param(self, params_data): 

272 for p, pdata in zip(self._params, params_data): 

273 p.data.copy_(pdata)

275 def _directional_evaluate(self, closure, x, t, d): 

276 self._add_grad(t, d) 

277 loss = float(closure()) 

278 flat_grad = self._gather_flat_grad() 

279 self._set_param(x) 

280 return loss, flat_grad

282 def step(self, closure): 

283 """Performs a single optimization step. 

284 

285 Arguments: 

286 closure (callable): A closure that reevaluates the model 

287 and returns the loss. 

288 """ 

289 assert len(self.param_groups) == 1 

290 

291 group = self.param_groups[0] 

292 lr = group['lr'] 

293 max_iter = group['max_iter'] 

294 max_eval = group['max_eval'] 

295 tolerance_grad = group['tolerance_grad'] 

296 tolerance_change = group['tolerance_change'] 

297 line_search_fn = group['line_search_fn'] 

298 history_size = group['history_size']

301 # the first param, because this helps with casting in load_state_dict 

302 state = self.state[self._params[0]] 

303 state.setdefault('func_evals', 0) 

304 state.setdefault('n_iter', 0) 

305 

306 # evaluate initial f(x) and df/dx 

307 orig_loss = closure() 

308 loss = float(orig_loss) 

309 current_evals = 1 

310 state['func_evals'] += 1 

311 

312 flat_grad = self._gather_flat_grad() 

313 opt_cond = flat_grad.abs().max() <= tolerance_grad 

314 

315 # optimal condition 

316 if opt_cond: 

317 return orig_loss 

318 

319 # tensors cached in state (for tracing) 

320 d = state.get('d') 

321 t = state.get('t') 

322 old_dirs = state.get('old_dirs') 

323 old_stps = state.get('old_stps') 

324 ro = state.get('ro') 

325 H_diag = state.get('H_diag') 

326 prev_flat_grad = state.get('prev_flat_grad') 

327 prev_loss = state.get('prev_loss') 

328 

329 n_iter = 0 

330 # optimize for a max of max_iter iterations 

331 while n_iter < max_iter: 

332 # keep track of nb of iterations 

333 n_iter += 1 

334 state['n_iter'] += 1 

335 

336 ############################################################ 

337 # compute gradient descent direction 

338 ############################################################ 

339 if state['n_iter'] == 1: 

340 d = flat_grad.neg() 

341 old_dirs = [] 

342 old_stps = [] 

343 ro = [] 

344 H_diag = 1 

345 else: 

346 # do lbfgs update (update memory) 

347 y = flat_grad.sub(prev_flat_grad) 

348 s = d.mul(t) 

349 ys = y.dot(s) # y*s 

350 if ys > 1e-10: 

351 # updating memory 

352 if len(old_dirs) == history_size: 

353 # shift history by one (limited-memory) 

354 old_dirs.pop(0) 

355 old_stps.pop(0) 

356 ro.pop(0) 

357 

358 # store new direction/step 

359 old_dirs.append(y) 

360 old_stps.append(s) 

361 ro.append(1. / ys) 

362 

363 # update scale of initial Hessian approximation 

364 H_diag = ys / y.dot(y) # (y*y) 

365 

366 # compute the approximate (L-BFGS) inverse Hessian 

367 # multiplied by the gradient 

368 num_old = len(old_dirs) 

369 

370 if 'al' not in state: 

371 state['al'] = [None] * history_size 

372 al = state['al'] 

373 

374 # iteration in L-BFGS loop collapsed to use just one buffer 

375 q = flat_grad.neg() 

376 for i in range(num_old - 1, -1, -1): 

377 al[i] = old_stps[i].dot(q) * ro[i] 

378 q.add_(-al[i], old_dirs[i]) 

379 

380 # multiply by initial Hessian 

381 # r/d is the final direction 

382 d = r = torch.mul(q, H_diag) 

383 for i in range(num_old): 

384 be_i = old_dirs[i].dot(r) * ro[i] 

385 r.add_(al[i] - be_i, old_stps[i]) 

386 

387 if prev_flat_grad is None: 

388 prev_flat_grad = flat_grad.clone(memory_format=torch.contiguous_format) 

389 else: 

390 prev_flat_grad.copy_(flat_grad) 

391 prev_loss = loss 

392 

393 ############################################################ 

394 # compute step length 

395 ############################################################ 

396 # reset initial guess for step size 

397 if state['n_iter'] == 1: 

398 t = min(1., 1. / flat_grad.abs().sum()) * lr 

399 else: 

400 t = lr 

401 

402 # directional derivative 

403 gtd = flat_grad.dot(d) # g * d 

404 

405 # directional derivative is below tolerance 

406 if gtd > -tolerance_change: 

407 break 

408 

409 # optional line search: user function 

410 ls_func_evals = 0 

411 if line_search_fn is not None: 

412 # perform line search, using user function 

413 if line_search_fn != "strong_wolfe": 

414 raise RuntimeError("only 'strong_wolfe' is supported") 

415 else: 

416 x_init = self._clone_param() 

417 

418 def obj_func(x, t, d): 

419 return self._directional_evaluate(closure, x, t, d) 

420 

421 loss, flat_grad, t, ls_func_evals = _strong_wolfe( 

422 obj_func, x_init, t, d, loss, flat_grad, gtd) 

423 self._add_grad(t, d) 

424 opt_cond = flat_grad.abs().max() <= tolerance_grad 

425 else: 

426 # no line search, simply move with fixed-step 

427 self._add_grad(t, d) 

428 if n_iter != max_iter: 

429 # re-evaluate function only if not in last iteration 

430 # the reason we do this: in a stochastic setting, 

431 # no use to re-evaluate that function here 

432 loss = float(closure()) 

433 flat_grad = self._gather_flat_grad() 

434 opt_cond = flat_grad.abs().max() <= tolerance_grad 

435 ls_func_evals = 1 

436 

437 # update func eval 

438 current_evals += ls_func_evals 

439 state['func_evals'] += ls_func_evals 

440 

441 ############################################################ 

442 # check conditions 

443 ############################################################ 

444 if n_iter == max_iter: 

445 break 

446 

447 if current_evals >= max_eval: 

448 break 

449 

450 # optimal condition 

451 if opt_cond: 

452 break 

453 

454 # lack of progress 

455 if d.mul(t).abs().max() <= tolerance_change: 

456 break 

457 

458 if abs(loss - prev_loss) < tolerance_change: 

459 break 

460 

461 state['d'] = d 

462 state['t'] = t 

463 state['old_dirs'] = old_dirs 

464 state['old_stps'] = old_stps 

465 state['ro'] = ro 

466 state['H_diag'] = H_diag 

467 state['prev_flat_grad'] = prev_flat_grad 

468 state['prev_loss'] = prev_loss 

469 

470 return orig_loss
