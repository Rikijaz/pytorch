14# Setup for the hypothesis tests. 

15# The tuples are (torch_quantized_dtype, zero_point_enforce), where the last 

16# element is enforced zero_point. If None, any zero_point point within the 

17# range of the data type is OK. 

18 

19# Tuple with all quantized data types. 

20_ALL_QINT_TYPES = ( 

21 torch.quint8, 

22 torch.qint8, 

23 torch.qint32, 

24)

26# Enforced zero point for every quantized data type. 

27# If None, any zero_point point within the range of the data type is OK. 

28_ENFORCED_ZERO_POINT = defaultdict(lambda: None, { 

29 torch.quint8: None, 

30 torch.qint8: None, 

31 torch.qint32: 0 

32})

34def _get_valid_min_max(qparams): 

35 scale, zero_point, quantized_type = qparams 

36 adjustment = 1 + torch.finfo(torch.float).eps 

37 _long_type_info = torch.iinfo(torch.long) 

38 long_min, long_max = _long_type_info.min / adjustment, _long_type_info.max / adjustment 

39 # make sure intermediate results are within the range of long 

40 min_value = max((long_min - zero_point) * scale, (long_min / scale + zero_point)) 

41 max_value = min((long_max - zero_point) * scale, (long_max / scale + zero_point)) 

42 return np.float32(min_value), np.float32(max_value)

44# This wrapper wraps around `st.floats` and checks the version of `hypothesis`, if 

45# it is too old, removes the `width` parameter (which was introduced) 

46# in 3.67.0 

47def _floats_wrapper(*args, **kwargs): 

48 if 'width' in kwargs and hypothesis.version.__version_info__ < (3, 67, 0): 

49 kwargs.pop('width') 

50 return st.floats(*args, **kwargs)

52def floats(*args, **kwargs):

54 kwargs['width'] = 32

57"""Hypothesis filter to avoid overflows with quantized tensors. 

58 

59Args: 

60 tensor: Tensor of floats to filter 

61 qparams: Quantization parameters as returned by the `qparams`. 

62 

63Returns: 

64 True 

65 

66Raises: 

67 hypothesis.UnsatisfiedAssumption 

68 

69Note: This filter is slow. Use it only when filtering of the test cases is 

70 absolutely necessary!

72def assume_not_overflowing(tensor, qparams): 

73 min_value, max_value = _get_valid_min_max(qparams) 

74 assume(tensor.min() >= min_value) 

75 assume(tensor.max() <= max_value) 

76 return True

78"""Strategy for generating the quantization parameters. 

79 

80Args: 

81 dtypes: quantized data types to sample from. 

82 scale_min / scale_max: Min and max scales. If None, set to 1e-3 / 1e3. 

83 zero_point_min / zero_point_max: Min and max for the zero point. If None, 

84 set to the minimum and maximum of the quantized data type. 

85 Note: The min and max are only valid if the zero_point is not enforced 

86 by the data type itself. 

87 

88Generates: 

89 scale: Sampled scale. 

90 zero_point: Sampled zero point. 

91 quantized_type: Sampled quantized type. 

92""" 

93@st.composite 

94def qparams(draw, dtypes=None, scale_min=None, scale_max=None, 

95 zero_point_min=None, zero_point_max=None): 

96 if dtypes is None: 

97 dtypes = _ALL_QINT_TYPES 

98 if not isinstance(dtypes, (list, tuple)): 

99 dtypes = (dtypes,) 

100 quantized_type = draw(st.sampled_from(dtypes)) 

101 

102 _type_info = torch.iinfo(quantized_type) 

103 qmin, qmax = _type_info.min, _type_info.max 

104 

105 # TODO: Maybe embed the enforced zero_point in the `torch.iinfo`. 

106 _zp_enforced = _ENFORCED_ZERO_POINT[quantized_type] 

107 if _zp_enforced is not None: 

108 zero_point = _zp_enforced 

109 else: 

110 _zp_min = qmin if zero_point_min is None else zero_point_min 

111 _zp_max = qmax if zero_point_max is None else zero_point_max 

112 zero_point = draw(st.integers(min_value=_zp_min, max_value=_zp_max)) 

113 

114 if scale_min is None: 

115 scale_min = torch.finfo(torch.float).eps 

116 if scale_max is None: 

117 scale_max = torch.finfo(torch.float).max 

118 scale = draw(floats(min_value=scale_min, max_value=scale_max, width=32)) 

119 

120 return scale, zero_point, quantized_type

149"""Strategy for generating test cases for tensors. 

150The resulting tensor is in float32 format. 

151 

152Args: 

153 shapes: Shapes under test for the tensor. Could be either a hypothesis 

154 strategy, or an iterable of different shapes to sample from. 

155 elements: Elements to generate from for the returned data type. 

156 If None, the strategy resolves to float within range [-1e6, 1e6]. 

157 qparams: Instance of the qparams strategy. This is used to filter the tensor 

158 such that the overflow would not happen. 

159 

160Generates: 

161 X: Tensor of type float32. Note that NaN and +/-inf is not included. 

162 qparams: (If `qparams` arg is set) Quantization parameters for X. 

163 The returned parameters are `(scale, zero_point, quantization_type)`. 

164 (If `qparams` arg is None), returns None. 

165""" 

166@st.composite 

167def tensor(draw, shapes=None, elements=None, qparams=None): 

168 if isinstance(shapes, SearchStrategy): 

169 _shape = draw(shapes) 

170 else: 

171 _shape = draw(st.sampled_from(shapes)) 

172 if qparams is None: 

173 if elements is None: 

174 elements = floats(-1e6, 1e6, allow_nan=False, width=32) 

175 X = draw(stnp.arrays(dtype=np.float32, elements=elements, shape=_shape)) 

176 assume(not (np.isnan(X).any() or np.isinf(X).any())) 

177 return X, None 

178 qparams = draw(qparams) 

179 if elements is None: 

180 min_value, max_value = _get_valid_min_max(qparams) 

181 elements = floats(min_value, max_value, allow_infinity=False, 

182 allow_nan=False, width=32) 

183 X = draw(stnp.arrays(dtype=np.float32, elements=elements, shape=_shape)) 

184 # Recompute the scale and zero_points according to the X statistics. 

185 scale, zp = _calculate_dynamic_qparams(X, qparams[2]) 

186 enforced_zp = _ENFORCED_ZERO_POINT.get(qparams[2], None) 

187 if enforced_zp is not None: 

188 zp = enforced_zp 

189 return X, (scale, zp, qparams[2])

218 permute_axes[axis] = 0 

219 X = np.transpose(X, permute_axes) 

220

223"""Strategy for generating test cases for tensors used in Conv. 

224The resulting tensors is in float32 format. 

225 

226Args: 

227 spatial_dim: Spatial Dim for feature maps. 

228 batch_size_range: Range to generate `batch_size`. 

229 Must be tuple of `(min, max)`. 

230 input_channels_per_group_range: 

231 Range to generate `input_channels_per_group`. 

232 Must be tuple of `(min, max)`. 

233 output_channels_per_group_range: 

234 Range to generate `output_channels_per_group`. 

235 Must be tuple of `(min, max)`. 

236 feature_map_range: Range to generate feature map size for each spatial_dim. 

237 Must be tuple of `(min, max)`. 

238 kernel_range: Range to generate kernel size for each spatial_dim. Must be 

239 tuple of `(min, max)`. 

240 max_groups: Maximum number of groups to generate. 

241 elements: Elements to generate from for the returned data type. 

242 If None, the strategy resolves to float within range [-1e6, 1e6]. 

243 qparams: Strategy for quantization parameters. for X, w, and b. 

244 Could be either a single strategy (used for all) or a list of 

245 three strategies for X, w, b. 

246Generates: 

247 (X, W, b, g): Tensors of type `float32` of the following drawen shapes: 

248 X: (`batch_size, input_channels, H, W`) 

249 W: (`output_channels, input_channels_per_group) + kernel_shape 

250 b: `(output_channels,)` 

251 groups: Number of groups the input is divided into 

252Note: X, W, b are tuples of (Tensor, qparams), where qparams could be either 

253 None or (scale, zero_point, quantized_type) 

254 

255 

256Example: 

257 @given(tensor_conv( 

258 spatial_dim=2, 

259 batch_size_range=(1, 3), 

260 input_channels_per_group_range=(1, 7), 

261 output_channels_per_group_range=(1, 7), 

262 feature_map_range=(6, 12), 

263 kernel_range=(3, 5), 

264 max_groups=4, 

265 elements=st.floats(-1.0, 1.0), 

266 qparams=qparams() 

267 )) 

268""" 

269@st.composite 

270def tensor_conv( 

271 draw, spatial_dim=2, batch_size_range=(1, 4), 

272 input_channels_per_group_range=(3, 7), 

273 output_channels_per_group_range=(3, 7), feature_map_range=(6, 12), 

274 kernel_range=(3, 7), max_groups=1, elements=None, qparams=None 

275): 

276 

277 # Resolve the minibatch, in_channels, out_channels, iH/iW, iK/iW 

278 batch_size = draw(st.integers(*batch_size_range)) 

279 input_channels_per_group = draw( 

280 st.integers(*input_channels_per_group_range)) 

281 output_channels_per_group = draw( 

282 st.integers(*output_channels_per_group_range)) 

283 groups = draw(st.integers(1, max_groups)) 

284 input_channels = input_channels_per_group * groups 

285 output_channels = output_channels_per_group * groups 

286 

287 feature_map_shape = [] 

288 for i in range(spatial_dim): 

289 feature_map_shape.append(draw(st.integers(*feature_map_range))) 

290 

291 kernels = [] 

292 for i in range(spatial_dim): 

293 kernels.append(draw(st.integers(*kernel_range))) 

294 

295 # Resolve the tensors 

296 if qparams is not None: 

297 if isinstance(qparams, (list, tuple)): 

298 assert(len(qparams) == 3), "Need 3 qparams for X, w, b" 

299 else: 

300 qparams = [qparams] * 3 

301 

302 X = draw(tensor(shapes=( 

303 (batch_size, input_channels) + tuple(feature_map_shape),), 

304 elements=elements, qparams=qparams[0])) 

305 W = draw(tensor(shapes=( 

306 (output_channels, input_channels_per_group) + tuple(kernels),), 

307 elements=elements, qparams=qparams[1])) 

308 b = draw(tensor(shapes=(output_channels,), elements=elements, 

309 qparams=qparams[2])) 

310 

311 return X, W, b, groups
