225class MinMaxObserver(_ObserverBase): 

226 r"""Observer module for computing the quantization parameters based on the 

227 running min and max values. 

228 

229 This observer uses the tensor min/max statistics to compute the quantization 

230 parameters. The module records the running minimum and maximum of incoming 

231 tensors, and uses this statistic to compute the quantization parameters. 

232 

233 Args: 

234 dtype: Quantized data type 

235 qscheme: Quantization scheme to be used 

236 reduce_range: Reduces the range of the quantized data type by 1 bit 

237 

238 Given running min/max as :math:`x_\text{min}` and :math:`x_\text{max}`, 

239 scale :math:`s` and zero point :math:`z` are computed as: 

240 

241 The running minimum/maximum :math:`x_\text{min/max}` is computed as: 

242 

243 .. math:: 

244 

245 \begin{array}{ll} 

246 x_\text{min} &= \begin{cases} 

247 \min(X) & \text{if~}x_\text{min} = \text{None} \\ 

248 \min\left(x_\text{min}, \min(X)\right) & \text{otherwise} 

249 \end{cases}\\ 

250 x_\text{max} &= \begin{cases} 

251 \max(X) & \text{if~}x_\text{max} = \text{None} \\ 

252 \max\left(x_\text{max}, \max(X)\right) & \text{otherwise} 

253 \end{cases}\\ 

254 \end{array} 

255 

256 where :math:`X` is the observed tensor. 

257 

258 The scale :math:`s` and zero point :math:`z` are then computed as: 

259 

260 .. math:: 

261 

262 \begin{aligned} 

263 \text{if Symmetric:}&\\ 

264 &s = 2 \max(|x_\text{min}|, x_\text{max}) / 

265 \left( Q_\text{max} - Q_\text{min} \right) \\ 

266 &z = \begin{cases} 

267 0 & \text{if dtype is qint8} \\ 

268 128 & \text{otherwise} 

269 \end{cases}\\ 

270 \text{Otherwise:}&\\ 

271 &s = \left( x_\text{max} - x_\text{min} \right ) / 

272 \left( Q_\text{max} - Q_\text{min} \right ) \\ 

273 &z = Q_\text{min} - \text{round}(x_\text{min} / s) 

274 \end{aligned} 

275 

276 where :math:`Q_\text{min}` and :math:`Q_\text{max}` are the minimum and 

277 maximum of the quantized data type. 

278 

279 .. warning:: Only works with ``torch.per_tensor_symmetric`` quantization scheme 

280 

281 .. warning:: :attr:`dtype` can only take ``torch.qint8`` or ``torch.quint8``. 

282 

283 .. note:: If the running minimum equals to the running maximum, the scale 

284 and zero_point are set to 1.0 and 0. 

285 """

289 # For x86 quantized kernels, we need to ensure that the vpmaddubsw 

290 # instruction does not overflow. We allow for a reduce_range argument to 

291 # observers that reduces the quantized range to (0,127) or (-64, 63). 

292 # For more details see aten/src/ATen/native/quantized/cpu/qconv.cpp 

293 # This is not an optimal choice for non x86 backends as it loses a bit 

294 # of precision for activations. 

295 

296 super(MinMaxObserver, self).__init__(dtype=dtype, 

297 qscheme=qscheme, 

298 reduce_range=reduce_range) 

299 self.min_val = torch.tensor([]) 

300 self.max_val = torch.tensor([]) 

301 if self.qscheme == torch.per_tensor_symmetric and \ 

302 self.reduce_range and \ 

303 self.dtype == torch.quint8: 

304 raise NotImplementedError("Cannot reduce range for symmetric \ 

305 quantization for quint8")

350class MovingAverageMinMaxObserver(MinMaxObserver): 

351 r"""Observer module for computing the quantization parameters based on the 

352 moving average of the min and max values. 

353 

354 This observer computes the quantization parameters based on the moving 

355 averages of minimums and maximums of the incoming tensors. The module 

356 records the average minimum and maximum of incoming tensors, and uses this 

357 statistic to compute the quantization parameters. 

358 

359 Args: 

360 averaging_constant: Averaging constant for min/max. 

361 dtype: Quantized data type 

362 qscheme: Quantization scheme to be used 

363 reduce_range: Reduces the range of the quantized data type by 1 bit 

364 

365 The moving average min/max is computed as follows 

366 

367 .. math:: 

368 

369 \begin{array}{ll} 

370 x_\text{min} = \begin{cases} 

371 \min(X) & \text{if~}x_\text{min} = \text{None} \\ 

372 (1 - c) x_\text{min} + c \min(X) & \text{otherwise} 

373 \end{cases}\\ 

374 x_\text{max} = \begin{cases} 

375 \max(X) & \text{if~}x_\text{max} = \text{None} \\ 

376 (1 - c) x_\text{max} + c \max(X) & \text{otherwise} 

377 \end{cases}\\ 

378 \end{array} 

379 

380 where :math:`x_\text{min/max}` is the running average min/max, :math:`X` is 

381 is the incoming tensor, and :math:`c` is the ``averaging_constant``. 

382 

383 The scale and zero point are then computed as in 

384 :class:`~torch.quantization.observer.MinMaxObserver`. 

385 

386 .. note:: Only works with ``torch.per_tensor_affine`` quantization shceme. 

387 

388 .. note:: If the running minimum equals to the running maximum, the scale 

389 and zero_point are set to 1.0 and 0. 

390 """ 

391 def __init__(self, averaging_constant=0.01, dtype=torch.quint8, 

392 qscheme=torch.per_tensor_affine, reduce_range=False): 

393 self.averaging_constant = averaging_constant 

394 super(MovingAverageMinMaxObserver, self).__init__(dtype=dtype, 

395 qscheme=qscheme, 

396 reduce_range=reduce_range) 

397 

398 def forward(self, x_orig): 

399 x = x_orig.detach() # avoid keeping autograd tape 

400 min_val = self.min_val 

401 max_val = self.max_val 

402 if min_val.numel() == 0 or max_val.numel() == 0: 

403 min_val = torch.min(x) 

404 max_val = torch.max(x) 

405 else: 

406 min_val = min_val + self.averaging_constant * (torch.min(x) - min_val) 

407 max_val = max_val + self.averaging_constant * (torch.max(x) - max_val) 

408 self.min_val = min_val 

409 self.max_val = max_val 

410 return x_orig

561class HistogramObserver(_ObserverBase): 

562 r""" 

563 The module records the running histogram of tensor values along with 

564 min/max values. ``calculate_qparams`` will calculate scale and zero_point. 

565 

566 Args: 

567 bins: Number of bins to use for the histogram 

568 upsample_rate: Factor by which the histograms are upsampled, this is 

569 used to interpolate histograms with varying ranges across observations 

570 dtype: Quantized data type 

571 qscheme: Quantization scheme to be used 

572 reduce_range: Reduces the range of the quantized data type by 1 bit 

573 

574 The scale and zero point are computed as follows: 

575 

576 1. Create the histogram of the incoming inputs. 

577 The histogram is computed continuously, and the ranges per bin change 

578 with every new tensor observed. 

579 2. Search the distribution in the histogram for optimal min/max values. 

580 The search for the min/max values ensures the minimization of the 

581 quantization error with respect to the floating point model. 

582 3. Compute the scale and zero point the same way as in the 

583 :class:`~torch.quantization.MinMaxObserver` 

584 """

588 # bins: The number of bins used for histogram calculation. 

589 super(HistogramObserver, self).__init__(dtype=dtype, 

590 qscheme=qscheme, 

591 reduce_range=reduce_range) 

592 self.bins = bins 

593 self.register_buffer('histogram', torch.zeros(self.bins)) 

594 self.min_val = torch.tensor([]) 

595 self.max_val = torch.tensor([]) 

596 self.dst_nbins = 2 ** torch.iinfo(self.dtype).bits 

597 self.upsample_rate = upsample_rate

601 r"""Non-linear parameter search. 

602 

603 An approximation for L2 error minimization for selecting min/max. 

604 By selecting new min/max, we filter out outliers in input distribution. 

605 This follows the implementation of NormMinimization::NonlinearQuantizationParamsSearch in 

606 caffe2/quantization/server/norm_minimization.cc 

607 """ 

608 def _get_norm(delta_begin, delta_end, density, norm_type): 

609 r""" 

610 Compute the norm of the values uniformaly distributed between 

611 delta_begin and delta_end. 

612 

613 norm = density * (integral_{begin, end} x^2) 

614 = density * (end^3 - begin^3) / 3 

615 """ 

616 assert norm_type == "L2", "Only L2 norms are currently supported" 

617 norm = 0.0 

618 if norm_type == "L2": 

619 norm = ( 

620 delta_end * delta_end * delta_end 

621 - delta_begin * delta_begin * delta_begin 

622 ) / 3 

623 return density * norm 

624 

625 def _compute_quantization_error(next_start_bin, next_end_bin, norm_type): 

626 r""" 

627 Compute the quantization error if we use start_bin to end_bin as the 

628 min and max to do the quantization. 

629 """ 

630 bin_width = (self.max_val.item() - self.min_val.item()) / self.bins 

631 

632 norm = 0.0 

633 dst_bin_width = bin_width * (next_end_bin - next_start_bin + 1) / self.dst_nbins 

634 if dst_bin_width == 0.0: 

635 return 0.0 

636 for src_bin in range(self.bins): 

637 # distances from the beginning of first dst_bin to the beginning and 

638 # end of src_bin 

639 src_bin_begin = (src_bin - next_start_bin) * bin_width 

640 src_bin_end = src_bin_begin + bin_width 

641 

642 # which dst_bins the beginning and end of src_bin belong to? 

643 dst_bin_of_begin = min( 

644 self.dst_nbins - 1, max(0.0, math.floor(src_bin_begin / dst_bin_width)) 

645 ) 

646 dst_bin_of_end = min( 

647 self.dst_nbins - 1, max(0.0, math.floor(src_bin_end / dst_bin_width)) 

648 ) 

649 dst_bin_of_begin_center = ( 

650 dst_bin_of_begin * dst_bin_width + dst_bin_width / 2 

651 ) 

652 

653 density = self.histogram[src_bin] / bin_width 

654 if dst_bin_of_begin == dst_bin_of_end: 

655 # if src_bin is entirely within 1 dst_bin 

656 delta_begin = src_bin_begin - dst_bin_of_begin_center 

657 delta_end = src_bin_end - dst_bin_of_begin_center 

658 norm = norm + _get_norm(delta_begin, delta_end, density, norm_type) 

659 else: 

660 delta_begin = src_bin_begin - dst_bin_of_begin_center 

661 delta_end = dst_bin_width / 2 

662 norm = norm + _get_norm(delta_begin, delta_end, density, norm_type) 

663 

664 norm = norm + (dst_bin_of_end - dst_bin_of_begin - 1) * _get_norm( 

665 -dst_bin_width / 2, dst_bin_width / 2, density, norm_type 

666 ) 

667 

668 dst_bin_of_end_center = ( 

669 dst_bin_of_end * dst_bin_width + dst_bin_width / 2 

670 ) 

671 

672 delta_begin = -dst_bin_width / 2 

673 delta_end = src_bin_end - dst_bin_of_end_center 

674 norm = norm + _get_norm(delta_begin, delta_end, density, norm_type) 

675 return norm 

676 

677 assert self.histogram.size()[0] == self.bins, "bins mistmatch" 

678 bin_width = (self.max_val - self.min_val) / self.bins 

679 

680 # cumulative sum 

681 total = sum(self.histogram) 

682 cSum = torch.cumsum(self.histogram, dim=0) 

683 

684 stepsize = 1e-5 # granularity 

685 alpha = 0.0 # lower bound 

686 beta = 1.0 # upper bound 

687 start_bin = 0 

688 end_bin = self.bins - 1 

689 norm_min = float("inf") 

690 

691 while alpha < beta: 

692 # Find the next step 

693 next_alpha = alpha + stepsize 

694 next_beta = beta - stepsize 

695 

696 # find the left and right bins between the quantile bounds 

697 l = start_bin 

698 r = end_bin 

699 while l < end_bin and cSum[l] < next_alpha * total: 

700 l = l + 1 

701 while r > start_bin and cSum[r] > next_beta * total: 

702 r = r - 1 

703 

704 # decide the next move 

705 next_start_bin = start_bin 

706 next_end_bin = end_bin 

707 if (l - start_bin) > (end_bin - r): 

708 # move the start bin 

709 next_start_bin = l 

710 alpha = next_alpha 

711 else: 

712 # move the end bin 

713 next_end_bin = r 

714 beta = next_beta 

715 

716 if next_start_bin == start_bin and next_end_bin == end_bin: 

717 continue 

718 

719 # calculate the quantization error using next_start_bin and next_end_bin 

720 norm = _compute_quantization_error(next_start_bin, next_end_bin, "L2") 

721 

722 if norm > norm_min: 

723 break 

724 norm_min = norm 

725 start_bin = next_start_bin 

726 end_bin = next_end_bin 

727 

728 new_min = self.min_val + bin_width * start_bin 

729 new_max = self.min_val + bin_width * (end_bin + 1) 

730 return new_min, new_max 

731 

732 @torch.jit.ignore 

733 def _adjust_min_max(self, combined_min, combined_max, upsample_rate): 

734 # type: (Tensor, Tensor, int) -> Tuple[Tensor, Tensor, int, int] 

735 # We ensure that: 

736 # (combined_max - combined_min)/(downsample_rate*Nbins) = (max - min)/(upsample_rate*Nbins) 

737 # This allows us to have a common grid of resolution s, where we can align 

738 # the input histogram 

739 # start_idx maps min_val to the histogram bin index. 

740 

741 hist_bin_width = (self.max_val - self.min_val) / (self.bins * upsample_rate) 

742 downsample_rate = torch.ceil((combined_max - combined_min) / (self.bins * hist_bin_width)).to(torch.int).item() 

743 e = downsample_rate * (self.bins * hist_bin_width) - (combined_max - combined_min) 

744 combined_max = combined_max + e / 2 

745 combined_min = combined_min - e / 2 

746 start_idx = torch.round((self.min_val - combined_min) / hist_bin_width).to(torch.int).item() 

747 return combined_min, combined_max, downsample_rate, start_idx 

748 

749 @torch.jit.ignore 

750 def _combine_histograms(self, orig_hist, new_hist, upsample_rate, downsample_rate, start_idx, Nbins): 

751 # type: (Tensor, Tensor, int, int, int, int) -> Tensor 

752 # First up-sample the histogram with new data by a factor of L 

753 # This creates an approximate probability density thats piecwise constant 

754 upsampled_histogram = new_hist.repeat_interleave(upsample_rate) 

755 # Now insert the upsampled histogram into the output 

756 # histogram, which is initialized with zeros. 

757 # The offset at which the histogram is introduced is determined 

758 # by the start index as the output histogram can cover a wider range 

759 histogram_with_output_range = torch.zeros((Nbins * downsample_rate)) 

760 histogram_with_output_range[start_idx:Nbins * upsample_rate + start_idx] = upsampled_histogram 

761 # Compute integral histogram, double precision is needed to ensure 

762 # that there are no overflows 

763 integral_histogram = torch.cumsum(histogram_with_output_range, 0, 

764 dtype=torch.double)[downsample_rate - 1 :: downsample_rate] 

765 # Finally perform interpolation 

766 shifted_integral_histogram = torch.zeros((Nbins)) 

767 shifted_integral_histogram[1:Nbins] = integral_histogram[0:-1] 

768 interpolated_histogram = (integral_histogram - shifted_integral_histogram) / upsample_rate 

769 orig_hist = orig_hist + interpolated_histogram.to(torch.float) 

770 return orig_hist

772 def forward(self, x_orig): 

773 # type: (Tensor) -> Tensor 

774 x = x_orig.detach() 

775 min_val = self.min_val 

776 max_val = self.max_val 

777 if min_val.numel() == 0 or max_val.numel() == 0: 

778 min_val = torch.min(x) 

779 max_val = torch.max(x) 

780 self.min_val = min_val 

781 self.max_val = max_val 

782 self.histogram = torch.histc(x, self.bins, min=min_val, max=max_val) 

783 else: 

784 new_min = torch.min(x) 

785 new_max = torch.max(x) 

786 combined_min = torch.min(new_min, min_val) 

787 combined_max = torch.max(new_max, max_val) 

788 # combine the existing histogram and new histogram into 1 histogram 

789 # We do this by first upsampling the histogram to a dense grid 

790 # and then downsampling the histogram efficiently 

791 combined_min, combined_max, downsample_rate, start_idx = \ 

792 self._adjust_min_max(combined_min, combined_max, self.upsample_rate) 

793 combined_histogram = torch.histc(x, self.bins, min=combined_min, max=combined_max) 

794 if combined_min == min_val and combined_max == max_val: 

795 combined_histogram += self.histogram 

796 else: 

797 combined_histogram = self._combine_histograms( 

798 combined_histogram, 

799 self.histogram, 

800 self.upsample_rate, 

801 downsample_rate, 

802 start_idx, 

803 self.bins) 

804 

805 self.histogram = combined_histogram 

806 self.min_val = combined_min 

807 self.max_val = combined_max 

808 return x
