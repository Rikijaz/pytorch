168def check_compiler_abi_compatibility(compiler): 

169 ''' 

170 Verifies that the given compiler is ABI-compatible with PyTorch. 

171 

172 Arguments: 

173 compiler (str): The compiler executable name to check (e.g. ``g++``). 

174 Must be executable in a shell process. 

175 

176 Returns: 

177 False if the compiler is (likely) ABI-incompatible with PyTorch, 

178 else True. 

179 ''' 

180 if not _is_binary_build(): 

181 return True 

182 if os.environ.get('TORCH_DONT_CHECK_COMPILER_ABI') in ['ON', '1', 'YES', 'TRUE', 'Y']: 

183 return True 

184 

185 # First check if the compiler is one of the expected ones for the particular platform. 

186 if not check_compiler_ok_for_platform(compiler): 

187 warnings.warn(WRONG_COMPILER_WARNING.format( 

188 user_compiler=compiler, 

189 pytorch_compiler=_accepted_compilers_for_platform()[0], 

190 platform=sys.platform)) 

191 return False 

192 

193 if sys.platform.startswith('darwin'): 

194 # There is no particular minimum version we need for clang, so we're good here. 

195 return True 

196 try: 

197 if sys.platform.startswith('linux'): 

198 minimum_required_version = MINIMUM_GCC_VERSION 

199 version = subprocess.check_output([compiler, '-dumpfullversion', '-dumpversion']) 

200 version = version.decode().strip().split('.') 

201 else: 

202 minimum_required_version = MINIMUM_MSVC_VERSION 

203 compiler_info = subprocess.check_output(compiler, stderr=subprocess.STDOUT) 

204 match = re.search(r'(\d+)\.(\d+)\.(\d+)', compiler_info.decode().strip()) 

205 version = (0, 0, 0) if match is None else match.groups() 

206 except Exception:

208 warnings.warn('Error checking compiler version for {}: {}'.format(compiler, error)) 

209 return False 

210 

211 if tuple(map(int, version)) >= minimum_required_version: 

212 return True 

213 

214 compiler = '{} {}'.format(compiler, ".".join(version)) 

215 warnings.warn(ABI_INCOMPATIBILITY_WARNING.format(compiler)) 

216 

217 return False

542 def get_ext_filename(self, ext_name): 

543 # Get the original shared library name. For Python 3, this name will be 

544 # suffixed with "<SOABI>.so", where <SOABI> will be something like 

545 # cpython-37m-x86_64-linux-gnu. On Python 2, there is no such ABI name. 

546 # The final extension, .so, would be .lib/.dll on Windows of course. 

547 ext_filename = super(BuildExtension, self).get_ext_filename(ext_name) 

548 # If `no_python_abi_suffix` is `True`, we omit the Python 3 ABI 

549 # component. This makes building shared libraries with setuptools that 

550 # aren't Python modules nicer. 

551 if self.no_python_abi_suffix and sys.version_info >= (3, 0): 

552 # The parts will be e.g. ["my_extension", "cpython-37m-x86_64-linux-gnu", "so"]. 

553 ext_filename_parts = ext_filename.split('.') 

554 # Omit the second to last element. 

555 without_abi = ext_filename_parts[:-2] + ext_filename_parts[-1:] 

556 ext_filename = '.'.join(without_abi) 

557 return ext_filename

756def load(name, 

757 sources, 

758 extra_cflags=None, 

759 extra_cuda_cflags=None, 

760 extra_ldflags=None, 

761 extra_include_paths=None, 

762 build_directory=None, 

763 verbose=False, 

764 with_cuda=None, 

765 is_python_module=True): 

766 ''' 

767 Loads a PyTorch C++ extension just-in-time (JIT). 

768 

769 To load an extension, a Ninja build file is emitted, which is used to 

770 compile the given sources into a dynamic library. This library is 

771 subsequently loaded into the current Python process as a module and 

772 returned from this function, ready for use. 

773 

774 By default, the directory to which the build file is emitted and the 

775 resulting library compiled to is ``<tmp>/torch_extensions/<name>``, where 

776 ``<tmp>`` is the temporary folder on the current platform and ``<name>`` 

777 the name of the extension. This location can be overridden in two ways. 

778 First, if the ``TORCH_EXTENSIONS_DIR`` environment variable is set, it 

779 replaces ``<tmp>/torch_extensions`` and all extensions will be compiled 

780 into subfolders of this directory. Second, if the ``build_directory`` 

781 argument to this function is supplied, it overrides the entire path, i.e. 

782 the library will be compiled into that folder directly. 

783 

784 To compile the sources, the default system compiler (``c++``) is used, 

785 which can be overridden by setting the ``CXX`` environment variable. To pass 

786 additional arguments to the compilation process, ``extra_cflags`` or 

787 ``extra_ldflags`` can be provided. For example, to compile your extension 

788 with optimizations, pass ``extra_cflags=['-O3']``. You can also use 

789 ``extra_cflags`` to pass further include directories. 

790 

791 CUDA support with mixed compilation is provided. Simply pass CUDA source 

792 files (``.cu`` or ``.cuh``) along with other sources. Such files will be 

793 detected and compiled with nvcc rather than the C++ compiler. This includes 

794 passing the CUDA lib64 directory as a library directory, and linking 

795 ``cudart``. You can pass additional flags to nvcc via 

796 ``extra_cuda_cflags``, just like with ``extra_cflags`` for C++. Various 

797 heuristics for finding the CUDA install directory are used, which usually 

798 work fine. If not, setting the ``CUDA_HOME`` environment variable is the 

799 safest option. 

800 

801 Args: 

802 name: The name of the extension to build. This MUST be the same as the 

803 name of the pybind11 module! 

804 sources: A list of relative or absolute paths to C++ source files. 

805 extra_cflags: optional list of compiler flags to forward to the build. 

806 extra_cuda_cflags: optional list of compiler flags to forward to nvcc 

807 when building CUDA sources. 

808 extra_ldflags: optional list of linker flags to forward to the build. 

809 extra_include_paths: optional list of include directories to forward 

810 to the build. 

811 build_directory: optional path to use as build workspace. 

812 verbose: If ``True``, turns on verbose logging of load steps. 

813 with_cuda: Determines whether CUDA headers and libraries are added to 

814 the build. If set to ``None`` (default), this value is 

815 automatically determined based on the existence of ``.cu`` or 

816 ``.cuh`` in ``sources``. Set it to `True`` to force CUDA headers 

817 and libraries to be included. 

818 is_python_module: If ``True`` (default), imports the produced shared 

819 library as a Python module. If ``False``, loads it into the process 

820 as a plain dynamic library. 

821 

822 Returns: 

823 If ``is_python_module`` is ``True``, returns the loaded PyTorch 

824 extension as a Python module. If ``is_python_module`` is ``False`` 

825 returns nothing (the shared library is loaded into the process as a side 

826 effect). 

827 

828 Example: 

829 >>> from torch.utils.cpp_extension import load 

830 >>> module = load( 

831 name='extension', 

832 sources=['extension.cpp', 'extension_kernel.cu'], 

833 extra_cflags=['-O2'], 

834 verbose=True) 

835 ''' 

836 return _jit_compile( 

837 name, 

838 [sources] if isinstance(sources, str) else sources, 

839 extra_cflags, 

840 extra_cuda_cflags, 

841 extra_ldflags, 

842 extra_include_paths, 

843 build_directory or _get_build_directory(name, verbose), 

844 verbose, 

845 with_cuda, 

846 is_python_module)

849def load_inline(name, 

850 cpp_sources, 

851 cuda_sources=None, 

852 functions=None, 

853 extra_cflags=None, 

854 extra_cuda_cflags=None, 

855 extra_ldflags=None, 

856 extra_include_paths=None, 

857 build_directory=None, 

858 verbose=False, 

859 with_cuda=None, 

860 is_python_module=True, 

861 with_pytorch_error_handling=True): 

862 ''' 

863 Loads a PyTorch C++ extension just-in-time (JIT) from string sources. 

864 

865 This function behaves exactly like :func:`load`, but takes its sources as 

866 strings rather than filenames. These strings are stored to files in the 

867 build directory, after which the behavior of :func:`load_inline` is 

868 identical to :func:`load`. 

869 

870 See `the 

871 tests <https://github.com/pytorch/pytorch/blob/master/test/test_cpp_extensions.py>`_ 

872 for good examples of using this function. 

873 

874 Sources may omit two required parts of a typical non-inline C++ extension: 

875 the necessary header includes, as well as the (pybind11) binding code. More 

876 precisely, strings passed to ``cpp_sources`` are first concatenated into a 

877 single ``.cpp`` file. This file is then prepended with ``#include 

878 <torch/extension.h>``. 

879 

880 Furthermore, if the ``functions`` argument is supplied, bindings will be 

881 automatically generated for each function specified. ``functions`` can 

882 either be a list of function names, or a dictionary mapping from function 

883 names to docstrings. If a list is given, the name of each function is used 

884 as its docstring. 

885 

886 The sources in ``cuda_sources`` are concatenated into a separate ``.cu`` 

887 file and prepended with ``torch/types.h``, ``cuda.h`` and 

888 ``cuda_runtime.h`` includes. The ``.cpp`` and ``.cu`` files are compiled 

889 separately, but ultimately linked into a single library. Note that no 

890 bindings are generated for functions in ``cuda_sources`` per se. To bind 

891 to a CUDA kernel, you must create a C++ function that calls it, and either 

892 declare or define this C++ function in one of the ``cpp_sources`` (and 

893 include its name in ``functions``). 

894 

895 See :func:`load` for a description of arguments omitted below. 

896 

897 Args: 

898 cpp_sources: A string, or list of strings, containing C++ source code. 

899 cuda_sources: A string, or list of strings, containing CUDA source code. 

900 functions: A list of function names for which to generate function 

901 bindings. If a dictionary is given, it should map function names to 

902 docstrings (which are otherwise just the function names). 

903 with_cuda: Determines whether CUDA headers and libraries are added to 

904 the build. If set to ``None`` (default), this value is 

905 automatically determined based on whether ``cuda_sources`` is 

906 provided. Set it to ``True`` to force CUDA headers 

907 and libraries to be included. 

908 with_pytorch_error_handling: Determines whether pytorch error and 

909 warning macros are handled by pytorch instead of pybind. To do 

910 this, each function ``foo`` is called via an intermediary ``_safe_foo`` 

911 function. This redirection might cause issues in obscure cases 

912 of cpp. This flag should be set to ``False`` when this redirect 

913 causes issues. 

914 

915 Example: 

916 >>> from torch.utils.cpp_extension import load_inline 

917 >>> source = \'\'\' 

918 at::Tensor sin_add(at::Tensor x, at::Tensor y) { 

919 return x.sin() + y.sin(); 

920 } 

921 \'\'\' 

922 >>> module = load_inline(name='inline_extension', 

923 cpp_sources=[source], 

924 functions=['sin_add']) 

925 ''' 

926 build_directory = build_directory or _get_build_directory(name, verbose) 

927 

928 if isinstance(cpp_sources, str): 

929 cpp_sources = [cpp_sources] 

930 cuda_sources = cuda_sources or [] 

931 if isinstance(cuda_sources, str): 

932 cuda_sources = [cuda_sources] 

933 

934 cpp_sources.insert(0, '#include <torch/extension.h>') 

935 

936 # Adds a new `_get_safe_version(foo)` function that returns a new function 

937 # that performs the same operation as `foo` but with pytorch error handling 

938 # macros. 

939 cpp_sources.append(SAFE_FUNCTION_DEFINITION) 

940 

941 # If `functions` is supplied, we create the pybind11 bindings for the user. 

942 # Here, `functions` is (or becomes, after some processing) a map from 

943 # function names to function docstrings. 

944 if functions is not None: 

945 module_def = [] 

946 module_def.append('PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {') 

947 if isinstance(functions, str): 

948 functions = [functions] 

949 if isinstance(functions, list): 

950 # Make the function docstring the same as the function name. 

951 functions = dict((f, f) for f in functions) 

952 elif not isinstance(functions, dict): 

953 raise ValueError( 

954 "Expected 'functions' to be a list or dict, but was {}".format( 

955 type(functions))) 

956 for function_name, docstring in functions.items(): 

957 if with_pytorch_error_handling: 

958 module_def.append('m.def("{0}", _get_safe_version({0}), "{1}");'.format( 

959 function_name, docstring)) 

960 else: 

961 module_def.append('m.def("{0}", {0}, "{1}");'.format(function_name, docstring)) 

962 module_def.append('}') 

963 cpp_sources += module_def 

964 

965 cpp_source_path = os.path.join(build_directory, 'main.cpp') 

966 with open(cpp_source_path, 'w') as cpp_source_file: 

967 cpp_source_file.write('\n'.join(cpp_sources)) 

968 

969 sources = [cpp_source_path] 

970 

971 if cuda_sources: 

972 cuda_sources.insert(0, '#include <torch/types.h>') 

973 cuda_sources.insert(1, '#include <cuda.h>') 

974 cuda_sources.insert(2, '#include <cuda_runtime.h>') 

975 

976 cuda_source_path = os.path.join(build_directory, 'cuda.cu') 

977 with open(cuda_source_path, 'w') as cuda_source_file: 

978 cuda_source_file.write('\n'.join(cuda_sources)) 

979 

980 sources.append(cuda_source_path) 

981 

982 return _jit_compile( 

983 name, 

984 sources, 

985 extra_cflags, 

986 extra_cuda_cflags, 

987 extra_ldflags, 

988 extra_include_paths, 

989 build_directory, 

990 verbose, 

991 with_cuda, 

992 is_python_module)

997 extra_cflags, 

998 extra_cuda_cflags, 

999 extra_ldflags, 

1000 extra_include_paths, 

1001 build_directory, 

1002 verbose, 

1003 with_cuda, 

1004 is_python_module): 

1005 old_version = JIT_EXTENSION_VERSIONER.get_version(name) 

1006 version = JIT_EXTENSION_VERSIONER.bump_version_if_changed( 

1007 name, 

1008 sources, 

1009 build_arguments=[extra_cflags, extra_cuda_cflags, extra_ldflags, extra_include_paths], 

1010 build_directory=build_directory, 

1011 with_cuda=with_cuda 

1012 )

1015 print('The input conditions for extension module {} have changed. '.format(name) +

1020 baton = FileBaton(os.path.join(build_directory, 'lock'))

1202def _get_cuda_arch_flags(cflags=None): 

1203 ''' 

1204 Determine CUDA arch flags to use. 

1205 

1206 For an arch, say "6.1", the added compile flag will be 

1207 ``-gencode=arch=compute_61,code=sm_61``. 

1208 For an added "+PTX", an additional 

1209 ``-gencode=arch=compute_xx,code=compute_xx`` is added. 

1210 

1211 See select_compute_arch.cmake for corresponding named and supported arches 

1212 when building with CMake. 

1213 ''' 

1214 # If cflags is given, there may already be user-provided arch flags in it 

1215 # (from `extra_compile_args`) 

1216 if cflags is not None: 

1217 for flag in cflags: 

1218 if 'arch' in flag: 

1219 return [] 

1220 

1221 # Note: keep combined names ("arch1+arch2") above single names, otherwise 

1222 # string replacement may not do the right thing 

1223 named_arches = collections.OrderedDict([ 

1224 ('Kepler+Tesla', '3.7'), 

1225 ('Kepler', '3.5+PTX'), 

1226 ('Maxwell+Tegra', '5.3'), 

1227 ('Maxwell', '5.0;5.2+PTX'), 

1228 ('Pascal', '6.0;6.1+PTX'), 

1229 ('Volta', '7.0+PTX'), 

1230 ('Turing', '7.5+PTX'), 

1231 ]) 

1232 

1233 supported_arches = ['3.5', '3.7', '5.0', '5.2', '5.3', '6.0', '6.1', '6.2', 

1234 '7.0', '7.2', '7.5'] 

1235 valid_arch_strings = supported_arches + [s + "+PTX" for s in supported_arches] 

1236 

1237 # The default is sm_30 for CUDA 9.x and 10.x 

1238 # First check for an env var (same as used by the main setup.py) 

1239 # Can be one or more architectures, e.g. "6.1" or "3.5;5.2;6.0;6.1;7.0+PTX" 

1240 # See cmake/Modules_CUDA_fix/upstream/FindCUDA/select_compute_arch.cmake 

1241 arch_list = os.environ.get('TORCH_CUDA_ARCH_LIST', None) 

1242 

1243 # If not given, determine what's needed for the GPU that can be found 

1244 if not arch_list: 

1245 capability = torch.cuda.get_device_capability() 

1246 arch_list = ['{}.{}'.format(capability[0], capability[1])] 

1247 else: 

1248 # Deal with lists that are ' ' separated (only deal with ';' after) 

1249 arch_list = arch_list.replace(' ', ';') 

1250 # Expand named arches 

1251 for named_arch, archval in named_arches.items(): 

1252 arch_list = arch_list.replace(named_arch, archval) 

1253 

1254 arch_list = arch_list.split(';') 

1255 

1256 flags = [] 

1257 for arch in arch_list: 

1258 if arch not in valid_arch_strings: 

1259 raise ValueError("Unknown CUDA arch ({}) or GPU not supported".format(arch)) 

1260 else: 

1261 num = arch[0] + arch[2] 

1262 flags.append('-gencode=arch=compute_{},code=sm_{}'.format(num, num)) 

1263 if arch.endswith('+PTX'): 

1264 flags.append('-gencode=arch=compute_{},code=compute_{}'.format(num, num)) 

1265 

1266 return list(set(flags))

1288def _run_ninja_build(build_directory, verbose, error_prefix): 

1289 try: 

1290 sys.stdout.flush() 

1291 sys.stderr.flush() 

1292 if sys.version_info >= (3, 5): 

1293 # Warning: don't pass stdout=None to subprocess.run to get output. 

1294 # subprocess.run assumes that sys.__stdout__ has not been modified and 

1295 # attempts to write to it by default. However, when we call _run_ninja_build 

1296 # from ahead-of-time cpp extensions, the following happens: 

1297 # 1) If the stdout encoding is not utf-8, setuptools detachs __stdout__. 

1298 # https://github.com/pypa/setuptools/blob/7e97def47723303fafabe48b22168bbc11bb4821/setuptools/dist.py#L1110 

1299 # (it probably shouldn't do this) 

1300 # 2) subprocess.run (on POSIX, with no stdout override) relies on 

1301 # __stdout__ not being detached: 

1302 # https://github.com/python/cpython/blob/c352e6c7446c894b13643f538db312092b351789/Lib/subprocess.py#L1214 

1303 # To work around this, we pass in the fileno directly and hope that 

1304 # it is valid. 

1305 stdout_fileno = 1 

1306 subprocess.run( 

1307 ['ninja', '-v'], 

1308 stdout=stdout_fileno if verbose else subprocess.PIPE, 

1309 stderr=subprocess.STDOUT, 

1310 cwd=build_directory, 

1311 check=True) 

1312 else: 

1313 subprocess.check_output( 

1314 ['ninja', '-v'], 

1315 stderr=subprocess.STDOUT, 

1316 cwd=build_directory) 

1317 except subprocess.CalledProcessError: 

1318 # Python 2 and 3 compatible way of getting the error object. 

1319 _, error, _ = sys.exc_info() 

1320 # error.output contains the stdout and stderr of the build attempt. 

1321 message = error_prefix 

1322 if hasattr(error, 'output') and error.output: 

1323 message += ": {}".format(error.output.decode())

1433def _write_ninja_file(path, 

1434 cflags, 

1435 post_cflags, 

1436 cuda_cflags, 

1437 cuda_post_cflags, 

1438 sources, 

1439 objects, 

1440 ldflags, 

1441 library_target, 

1442 with_cuda): 

1443 """Write a ninja file that does the desired compiling and linking. 

1444 

1445 `path`: Where to write this file 

1446 `cflags`: list of flags to pass to $cxx. Can be None. 

1447 `post_cflags`: list of flags to append to the $cxx invocation. Can be None. 

1448 `cuda_cflags`: list of flags to pass to $nvcc. Can be None. 

1449 `cuda_postflags`: list of flags to append to the $nvcc invocation. Can be None. 

1450 `sources`: list of paths to source files 

1451 `objects`: list of desired paths to objects, one per source. 

1452 `ldflags`: list of flags to pass to linker. Can be None. 

1453 `library_target`: Name of the output library. Can be None; in that case, 

1454 we do no linking. 

1455 `with_cuda`: If we should be compiling with CUDA. 

1456 """ 

1457 def sanitize_flags(flags): 

1458 if flags is None: 

1459 return [] 

1460 else: 

1461 return [flag.strip() for flag in flags] 

1462 

1463 cflags = sanitize_flags(cflags) 

1464 post_cflags = sanitize_flags(post_cflags) 

1465 cuda_cflags = sanitize_flags(cuda_cflags) 

1466 cuda_post_cflags = sanitize_flags(cuda_post_cflags) 

1467 ldflags = sanitize_flags(ldflags) 

1468 

1469 # Sanity checks... 

1470 assert len(sources) == len(objects) 

1471 assert len(sources) > 0 

1472 

1473 if IS_WINDOWS: 

1474 compiler = os.environ.get('CXX', 'cl') 

1475 else: 

1476 compiler = os.environ.get('CXX', 'c++') 

1477 

1478 # Version 1.3 is required for the `deps` directive. 

1479 config = ['ninja_required_version = 1.3'] 

1480 config.append('cxx = {}'.format(compiler)) 

1481 if with_cuda: 

1482 config.append('nvcc = {}'.format(_join_cuda_home('bin', 'nvcc'))) 

1483 

1484 flags = ['cflags = {}'.format(' '.join(cflags))] 

1485 flags.append('post_cflags = {}'.format(' '.join(post_cflags))) 

1486 if with_cuda: 

1487 flags.append('cuda_cflags = {}'.format(' '.join(cuda_cflags))) 

1488 flags.append('cuda_post_cflags = {}'.format(' '.join(cuda_post_cflags))) 

1489 flags.append('ldflags = {}'.format(' '.join(ldflags))) 

1490 

1491 # Turn into absolute paths so we can emit them into the ninja build 

1492 # file wherever it is. 

1493 sources = [os.path.abspath(file) for file in sources] 

1494 

1495 # See https://ninja-build.org/build.ninja.html for reference. 

1496 compile_rule = ['rule compile'] 

1497 if IS_WINDOWS: 

1498 compile_rule.append( 

1499 ' command = cl /showIncludes $cflags -c $in /Fo$out $post_cflags') 

1500 compile_rule.append(' deps = msvc') 

1501 else: 

1502 compile_rule.append( 

1503 ' command = $cxx -MMD -MF $out.d $cflags -c $in -o $out $post_cflags') 

1504 compile_rule.append(' depfile = $out.d') 

1505 compile_rule.append(' deps = gcc') 

1506 

1507 if with_cuda: 

1508 cuda_compile_rule = ['rule cuda_compile'] 

1509 cuda_compile_rule.append( 

1510 ' command = $nvcc $cuda_cflags -c $in -o $out $cuda_post_cflags') 

1511 

1512 # Emit one build rule per source to enable incremental build. 

1513 build = [] 

1514 for source_file, object_file in zip(sources, objects): 

1515 is_cuda_source = _is_cuda_file(source_file) and with_cuda 

1516 rule = 'cuda_compile' if is_cuda_source else 'compile' 

1517 if IS_WINDOWS: 

1518 source_file = source_file.replace(':', '$:') 

1519 object_file = object_file.replace(':', '$:') 

1520 source_file = source_file.replace(" ", "$ ") 

1521 build.append('build {}: {} {}'.format(object_file, rule, source_file)) 

1522 

1523 if library_target is not None: 

1524 link_rule = ['rule link'] 

1525 if IS_WINDOWS: 

1526 cl_paths = subprocess.check_output(['where', 

1527 'cl']).decode().split('\r\n') 

1528 if len(cl_paths) >= 1: 

1529 cl_path = os.path.dirname(cl_paths[0]).replace(':', '$:') 

1530 else: 

1531 raise RuntimeError("MSVC is required to load C++ extensions") 

1532 link_rule.append( 

1533 ' command = "{}/link.exe" $in /nologo $ldflags /out:$out'.format( 

1534 cl_path)) 

1535 else: 

1536 link_rule.append(' command = $cxx $in $ldflags -o $out') 

1537 

1538 link = ['build {}: link {}'.format(library_target, ' '.join(objects))] 

1539 

1540 default = ['default {}'.format(library_target)] 

1541 else: 

1542 link_rule, link, default = [], [], [] 

1543 

1544 # 'Blocks' should be separated by newlines, for visual benefit. 

1545 blocks = [config, flags, compile_rule] 

1546 if with_cuda: 

1547 blocks.append(cuda_compile_rule) 

1548 blocks += [link_rule, build, link, default] 

1549 with open(path, 'w') as build_file: 

1550 for block in blocks: 

1551 lines = '\n'.join(block) 

1552 build_file.write('{}\n\n'.format(lines))
