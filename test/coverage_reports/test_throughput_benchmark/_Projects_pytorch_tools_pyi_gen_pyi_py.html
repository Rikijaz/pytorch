13""" 

14This module implements generation of type stubs for PyTorch, 

15enabling use of autocomplete in IDEs like PyCharm, which otherwise 

16don't understand C extension modules. 

17 

18At the moment, this module only handles type stubs for torch and 

19torch.Tensor. It should eventually be expanded to cover all functions 

20which come are autogenerated. 

21 

22Here's our general strategy: 

23 

24- We start off with a hand-written __init__.pyi.in file. This 

25 file contains type definitions for everything we cannot automatically 

26 generate, including pure Python definitions directly in __init__.py 

27 (the latter case should be pretty rare). 

28 

29- We go through automatically bound functions based on the 

30 type information recorded in Declarations.yaml and 

31 generate type hints for them (generate_type_hints) 

32 

33There are a number of type hints which we've special-cased; 

34read gen_pyi for the gory details. 

35"""

97def type_to_python(typename, size=None): 

98 """type_to_python(typename: str, size: str) -> str 

99 

100 Transforms a Declarations.yaml type name into a Python type specification 

101 as used for type hints. 

102 """ 

103 typename = typename.replace(' ', '') # normalize spaces, e.g., 'Generator *' 

104 

105 # Disambiguate explicitly sized int/tensor lists from implicitly 

106 # sized ones. These permit non-list inputs too. (IntArrayRef[] and 

107 # TensorList[] are not real types; this is just for convenience.) 

108 if typename in {'IntArrayRef', 'TensorList'} and size is not None: 

109 typename += '[]' 

110 

111 typename = { 

112 'Device': 'Union[_device, str, None]', 

113 'Generator*': 'Generator', 

114 'IntegerTensor': 'Tensor', 

115 'Scalar': 'Number', 

116 'ScalarType': '_dtype', 

117 'Storage': 'Storage', 

118 'BoolTensor': 'Tensor', 

119 'IndexTensor': 'Tensor', 

120 'Tensor': 'Tensor', 

121 'MemoryFormat': 'memory_format', 

122 'IntArrayRef': '_size', 

123 'IntArrayRef[]': 'Union[_int, _size]', 

124 'TensorList': 'Union[Tuple[Tensor, ...], List[Tensor]]', 

125 'TensorList[]': 'Union[Tensor, Tuple[Tensor, ...], List[Tensor]]', 

126 'bool': '_bool', 

127 'double': '_float', 

128 'int64_t': '_int', 

129 'accreal': 'Number', 

130 'real': 'Number', 

131 'void*': '_int', # data_ptr 

132 'void': 'None', 

133 'std::string': 'str', 

134 'Dimname': 'Union[str, None]', 

135 'DimnameList': 'List[Union[str, None]]', 

136 'QScheme': '_qscheme', 

137 }[typename] 

138 

139 return typename

142def arg_to_type_hint(arg): 

143 """arg_to_type_hint(arg) -> str 

144 

145 This takes one argument in a Declarations and returns a string 

146 representing this argument in a type hint signature. 

147 """ 

148 name = arg['name'] 

149 if name == 'from': # from is a Python keyword... 

150 name += '_' 

151 typename = type_to_python(arg['dynamic_type'], arg.get('size')) 

152 if arg.get('is_nullable'): 

153 typename = 'Optional[' + typename + ']' 

154 if 'default' in arg: 

155 default = arg['default'] 

156 if default == 'nullptr': 

157 default = None 

158 elif default == 'c10::nullopt': 

159 default = None 

160 elif isinstance(default, str) and default.startswith('{') and default.endswith('}'): 

161 if arg['dynamic_type'] == 'Tensor' and default == '{}': 

162 default = None 

163 elif arg['dynamic_type'] == 'IntArrayRef': 

164 default = '(' + default[1:-1] + ')' 

165 else: 

166 raise Exception("Unexpected default constructor argument of type {}".format(arg['dynamic_type'])) 

167 elif default == 'MemoryFormat::Contiguous': 

168 default = 'contiguous_format' 

169 elif default == 'QScheme::PER_TENSOR_AFFINE': 

170 default = 'per_tensor_affine' 

171 default = '={}'.format(default) 

172 else: 

173 default = '' 

174 return name + ': ' + typename + default

190def sig_for_ops(opname): 

191 """sig_for_ops(opname : str) -> List[str] 

192 

193 Returns signatures for operator special functions (__add__ etc.)""" 

194 

195 # we have to do this by hand, because they are hand-bound in Python 

196 

197 assert opname.endswith('__') and opname.startswith('__'), "Unexpected op {}".format(opname) 

198 

199 name = opname[2:-2] 

200 if name in binary_ops: 

201 return ['def {}(self, other: Any) -> Tensor: ...'.format(opname)] 

202 elif name in comparison_ops: 

203 # unsafe override https://github.com/python/mypy/issues/5704 

204 return ['def {}(self, other: Any) -> Tensor: ... # type: ignore'.format(opname)] 

205 elif name in unary_ops: 

206 return ['def {}(self) -> Tensor: ...'.format(opname)] 

207 elif name in to_py_type_ops: 

208 if name in {'bool', 'float'}: 

209 tname = name 

210 elif name == 'nonzero': 

211 tname = 'bool' 

212 else: 

213 tname = 'int' 

214 if tname in {'float', 'int', 'bool'}: 

215 tname = 'builtins.' + tname 

216 return ['def {}(self) -> {}: ...'.format(opname, tname)] 

217 else: 

218 raise Exception("unknown op", opname)

221def generate_type_hints(fname, decls, is_tensor=False): 

222 """generate_type_hints(fname, decls, is_tensor=False) 

223 

224 Generates type hints for the declarations pertaining to the function 

225 :attr:`fname`. attr:`decls` are the declarations from the parsed 

226 Declarations.yaml. 

227 The :attr:`is_tensor` flag indicates whether we are parsing 

228 members of the Tensor class (true) or functions in the 

229 `torch` namespace (default, false). 

230 

231 This function currently encodes quite a bit about the semantics of 

232 the translation C++ -> Python. 

233 """ 

234 if fname in blacklist: 

235 return [] 

236 

237 type_hints = [] 

238 dnames = ([d['name'] for d in decls]) 

239 has_out = fname + '_out' in dnames 

240 

241 if has_out: 

242 decls = [d for d in decls if d['name'] != fname + '_out'] 

243 

244 for decl in decls: 

245 render_kw_only_separator = True # whether we add a '*' if we see a keyword only argument 

246 python_args = [] 

247 

248 has_tensor_options = 'TensorOptions' in (a['dynamic_type'] for a in decl['arguments']) 

249 

250 for a in decl['arguments']: 

251 if a['dynamic_type'] != 'TensorOptions': 

252 if a.get('kwarg_only', False) and render_kw_only_separator: 

253 python_args.append('*') 

254 render_kw_only_separator = False 

255 try: 

256 python_args.append(arg_to_type_hint(a)) 

257 except Exception: 

258 print("Error while processing function {}".format(fname)) 

259 raise 

260 

261 if is_tensor: 

262 if 'self: Tensor' in python_args: 

263 python_args.remove('self: Tensor') 

264 python_args = ['self'] + python_args 

265 else: 

266 raise Exception("method without self is unexpected") 

267 

268 if has_out: 

269 if render_kw_only_separator: 

270 python_args.append('*') 

271 render_kw_only_separator = False 

272 python_args.append('out: Optional[Tensor]=None') 

273 

274 if has_tensor_options: 

275 if render_kw_only_separator: 

276 python_args.append('*') 

277 render_kw_only_separator = False 

278 python_args += ["dtype: _dtype=None", 

279 "layout: _layout=strided", 

280 "device: Union[_device, str, None]=None", 

281 "requires_grad:_bool=False"] 

282 

283 python_args_s = ', '.join(python_args) 

284 python_returns = [type_to_python(r['dynamic_type']) for r in decl['returns']] 

285 

286 if len(python_returns) > 1: 

287 python_returns_s = 'Tuple[' + ', '.join(python_returns) + ']' 

288 elif len(python_returns) == 1: 

289 python_returns_s = python_returns[0] 

290 else: 

291 python_returns_s = 'None' 

292 

293 type_hint = "def {}({}) -> {}: ...".format(fname, python_args_s, python_returns_s) 

294 numargs = len(decl['arguments']) 

295 vararg_pos = int(is_tensor) 

296 have_vararg_version = (numargs > vararg_pos and 

297 decl['arguments'][vararg_pos]['dynamic_type'] in {'IntArrayRef', 'TensorList'} and 

298 (numargs == vararg_pos + 1 or python_args[vararg_pos + 1] == '*') and 

299 (not is_tensor or decl['arguments'][0]['name'] == 'self')) 

300 

301 type_hints.append(type_hint) 

302 

303 if have_vararg_version: 

304 # Two things come into play here: PyTorch has the "magic" that if the first and only positional argument 

305 # is an IntArrayRef or TensorList, it will be used as a vararg variant. 

306 # The following outputs the vararg variant, the "pass a list variant" is output above. 

307 # The other thing is that in Python, the varargs are annotated with the element type, not the list type. 

308 typelist = decl['arguments'][vararg_pos]['dynamic_type'] 

309 if typelist == 'IntArrayRef': 

310 vararg_type = '_int' 

311 else: 

312 vararg_type = 'Tensor' 

313 # replace first argument and eliminate '*' if present 

314 python_args = ((['self'] if is_tensor else []) + ['*' + decl['arguments'][vararg_pos]['name'] + 

315 ': ' + vararg_type] + python_args[vararg_pos + 2:]) 

316 python_args_s = ', '.join(python_args) 

317 type_hint = "def {}({}) -> {}: ...".format(fname, python_args_s, python_returns_s) 

318 type_hints.append(type_hint) 

319 

320 return type_hints

322def gen_nn_modules(out): 

323 def replace_forward(m): 

324 # We instruct mypy to not emit errors for the `forward` and `__call__` declarations since mypy 

325 # would otherwise correctly point out that Module's descendants' `forward` declarations 

326 # conflict with `Module`s. Specifically, `Module` defines `forward(self, *args)` while the 

327 # descandantes define more specific forms, such as `forward(self, input: Tensor)`, which 

328 # violates Liskov substitutability. The 'mypy' team recommended this solution for now. 

329 forward_def = m.group(0) + " # type: ignore" 

330 call_def = re.sub(r'def forward', 'def __call__', forward_def) 

331 new_def = "{}\n{}".format(forward_def, call_def) 

332 return new_def 

333 pattern = re.compile(r'^\s*def forward\(self.*$', re.MULTILINE) 

334 for fname in glob.glob("torch/nn/modules/*.pyi.in"): 

335 with open(fname, 'r') as f: 

336 src = f.read() 

337 res = pattern.sub(replace_forward, src) 

338 fname_out = fname[:-3] 

339 with open(os.path.join(out, fname_out), 'w') as f: 

340 f.write(res)
