1""" 

2Python implementation of __torch_function__ 

3 

4While most of the torch API and handling for __torch_function__ happens 

5at the C++ level, some of the torch API is written in Python so we need

19def _get_overloaded_args(relevant_args): 

20 """Returns a list of arguments on which to call __torch_function__. 

21 

22 Checks arguments in relevant_args for __torch_function__ implementations, 

23 storing references to the arguments and their types in overloaded_args and 

24 overloaded_types in order of calling precedence. Only distinct types are 

25 considered. If a type is a subclass of another type it will have higher 

26 precedence, otherwise the precedence order is the same as the order of 

27 arguments in relevant_args, that is, from left-to-right in the argument list. 

28 

29 The precedence-determining algorithm implemented in this function is 

30 described in `NEP-0018`_. 

31 

32 See torch::append_overloaded_arg for the equivalent function in the C++ 

33 implementation. 

34 

35 Parameters 

36 ---------- 

37 relevant_args : iterable of array-like 

38 Iterable of array-like arguments to check for __torch_function__ 

39 methods. 

40 

41 Returns 

42 ------- 

43 overloaded_types : collection of types 

44 Types of arguments from relevant_args with __torch_function__ methods. 

45 overloaded_args : list 

46 Arguments from relevant_args on which to call __torch_function__ 

47 methods, in the order in which they should be called. 

48 

49 .. _NEP-0018: 

50 https://numpy.org/neps/nep-0018-array-function-protocol.html 

51 

52 """ 

53 # Runtime is O(num_arguments * num_unique_types) 

54 overloaded_types = [] 

55 overloaded_args = [] 

56 for arg in relevant_args: 

57 arg_type = type(arg) 

58 # We only collect arguments if they have a unique type, which ensures 

59 # reasonable performance even with a long list of possibly overloaded 

60 # arguments. 

61 if (arg_type not in overloaded_types and hasattr(arg_type, '__torch_function__')): 

62 # Create lists explicitly for the first type (usually the only one 

63 # done) to avoid setting up the iterator for overloaded_args. 

64 if overloaded_types: 

65 overloaded_types.append(arg_type) 

66 # By default, insert argument at the end, but if it is 

67 # subclass of another argument, insert it before that argument. 

68 # This ensures "subclasses before superclasses". 

69 index = len(overloaded_args) 

70 for i, old_arg in enumerate(overloaded_args): 

71 if issubclass(arg_type, type(old_arg)): 

72 index = i 

73 break 

74 overloaded_args.insert(index, arg) 

75 else: 

76 overloaded_types = [arg_type] 

77 overloaded_args = [arg] 

78 

79 return overloaded_args

82def handle_torch_function( 

83 public_api, relevant_args, *args, **kwargs): 

84 """Implement a function with checks for __torch_function__ overrides. 

85 

86 See torch::autograd::handle_torch_function for the equivalent of this 

87 function in the C++ implementation. 

88 

89 Arguments 

90 --------- 

91 public_api : function 

92 Function exposed by the public torch API originally called like 

93 ``public_api(*args, **kwargs)`` on which arguments are now being 

94 checked. 

95 relevant_args : iterable 

96 Iterable of arguments to check for __torch_function__ methods. 

97 args : tuple 

98 Arbitrary positional arguments originally passed into ``public_api``. 

99 kwargs : tuple 

100 Arbitrary keyword arguments originally passed into ``public_api``. 

101 

102 Returns 

103 ------- 

104 Result from calling `implementation()` or an `__torch_function__` 

105 method, as appropriate. 

106 

107 Raises 

108 ------ 

109 TypeError : if no implementation is found. 

110 

111 """ 

112 # Check for __torch_function__ methods. 

113 overloaded_args = _get_overloaded_args(relevant_args) 

114 

115 # Call overrides 

116 for overloaded_arg in overloaded_args: 

117 # Use `public_api` instead of `implementation` so __torch_function__ 

118 # implementations can do equality/identity comparisons. 

119 result = overloaded_arg.__torch_function__(public_api, args, kwargs) 

120 

121 if result is not NotImplemented: 

122 return result 

123 

124 func_name = '{}.{}'.format(public_api.__module__, public_api.__name__) 

125 raise TypeError("no implementation found for '{}' on types that implement " 

126 '__torch_function__: {}' 

127 .format(func_name, list(map(type, overloaded_args))))
