10class LocalResponseNorm(Module): 

11 r"""Applies local response normalization over an input signal composed 

12 of several input planes, where channels occupy the second dimension. 

13 Applies normalization across channels. 

14 

15 .. math:: 

16 b_{c} = a_{c}\left(k + \frac{\alpha}{n} 

17 \sum_{c'=\max(0, c-n/2)}^{\min(N-1,c+n/2)}a_{c'}^2\right)^{-\beta} 

18 

19 Args: 

20 size: amount of neighbouring channels used for normalization 

21 alpha: multiplicative factor. Default: 0.0001 

22 beta: exponent. Default: 0.75 

23 k: additive factor. Default: 1 

24 

25 Shape: 

26 - Input: :math:`(N, C, *)` 

27 - Output: :math:`(N, C, *)` (same shape as input) 

28 

29 Examples:: 

30 

31 >>> lrn = nn.LocalResponseNorm(2) 

32 >>> signal_2d = torch.randn(32, 5, 24, 24) 

33 >>> signal_4d = torch.randn(16, 5, 7, 7, 7, 7) 

34 >>> output_2d = lrn(signal_2d) 

35 >>> output_4d = lrn(signal_4d) 

36 

37 """ 

38 __constants__ = ['size', 'alpha', 'beta', 'k'] 

39 

40 def __init__(self, size, alpha=1e-4, beta=0.75, k=1.):

42 self.size = size 

43 self.alpha = alpha 

44 self.beta = beta 

45 self.k = k

72class LayerNorm(Module): 

73 r"""Applies Layer Normalization over a mini-batch of inputs as described in 

74 the paper `Layer Normalization`_ . 

75 

76 .. math:: 

77 y = \frac{x - \mathrm{E}[x]}{ \sqrt{\mathrm{Var}[x] + \epsilon}} * \gamma + \beta 

78 

79 The mean and standard-deviation are calculated separately over the last 

80 certain number dimensions which have to be of the shape specified by 

81 :attr:`normalized_shape`. 

82 :math:`\gamma` and :math:`\beta` are learnable affine transform parameters of 

83 :attr:`normalized_shape` if :attr:`elementwise_affine` is ``True``. 

84 

85 .. note:: 

86 Unlike Batch Normalization and Instance Normalization, which applies 

87 scalar scale and bias for each entire channel/plane with the 

88 :attr:`affine` option, Layer Normalization applies per-element scale and 

89 bias with :attr:`elementwise_affine`. 

90 

91 This layer uses statistics computed from input data in both training and 

92 evaluation modes. 

93 

94 Args: 

95 normalized_shape (int or list or torch.Size): input shape from an expected input 

96 of size 

97 

98 .. math:: 

99 [* \times \text{normalized\_shape}[0] \times \text{normalized\_shape}[1] 

100 \times \ldots \times \text{normalized\_shape}[-1]] 

101 

102 If a single integer is used, it is treated as a singleton list, and this module will 

103 normalize over the last dimension which is expected to be of that specific size. 

104 eps: a value added to the denominator for numerical stability. Default: 1e-5 

105 elementwise_affine: a boolean value that when set to ``True``, this module 

106 has learnable per-element affine parameters initialized to ones (for weights) 

107 and zeros (for biases). Default: ``True``. 

108 

109 Shape: 

110 - Input: :math:`(N, *)` 

111 - Output: :math:`(N, *)` (same shape as input) 

112 

113 Examples:: 

114 

115 >>> input = torch.randn(20, 5, 10, 10) 

116 >>> # With Learnable Parameters 

117 >>> m = nn.LayerNorm(input.size()[1:]) 

118 >>> # Without Learnable Parameters 

119 >>> m = nn.LayerNorm(input.size()[1:], elementwise_affine=False) 

120 >>> # Normalize over last two dimensions 

121 >>> m = nn.LayerNorm([10, 10]) 

122 >>> # Normalize over last dimension of size 10 

123 >>> m = nn.LayerNorm(10) 

124 >>> # Activating the module 

125 >>> output = m(input) 

126 

127 .. _`Layer Normalization`: https://arxiv.org/abs/1607.06450 

128 """ 

129 __constants__ = ['normalized_shape', 'eps', 'elementwise_affine'] 

130 

131 def __init__(self, normalized_shape, eps=1e-5, elementwise_affine=True): 

132 super(LayerNorm, self).__init__() 

133 if isinstance(normalized_shape, numbers.Integral): 

134 normalized_shape = (normalized_shape,) 

135 self.normalized_shape = tuple(normalized_shape) 

136 self.eps = eps 

137 self.elementwise_affine = elementwise_affine 

138 if self.elementwise_affine: 

139 self.weight = Parameter(torch.Tensor(*normalized_shape)) 

140 self.bias = Parameter(torch.Tensor(*normalized_shape)) 

141 else: 

142 self.register_parameter('weight', None) 

143 self.register_parameter('bias', None) 

144 self.reset_parameters() 

145 

146 def reset_parameters(self): 

147 if self.elementwise_affine: 

148 init.ones_(self.weight) 

149 init.zeros_(self.bias) 

150 

151 def forward(self, input): 

152 return F.layer_norm( 

153 input, self.normalized_shape, self.weight, self.bias, self.eps) 

154 

155 def extra_repr(self): 

156 return '{normalized_shape}, eps={eps}, ' \ 

157 'elementwise_affine={elementwise_affine}'.format(**self.__dict__)

160class GroupNorm(Module): 

161 r"""Applies Group Normalization over a mini-batch of inputs as described in 

162 the paper `Group Normalization`_ . 

163 

164 .. math:: 

165 y = \frac{x - \mathrm{E}[x]}{ \sqrt{\mathrm{Var}[x] + \epsilon}} * \gamma + \beta 

166 

167 The input channels are separated into :attr:`num_groups` groups, each containing 

168 ``num_channels / num_groups`` channels. The mean and standard-deviation are calculated 

169 separately over the each group. :math:`\gamma` and :math:`\beta` are learnable 

170 per-channel affine transform parameter vectors of size :attr:`num_channels` if 

171 :attr:`affine` is ``True``. 

172 

173 This layer uses statistics computed from input data in both training and 

174 evaluation modes. 

175 

176 Args: 

177 num_groups (int): number of groups to separate the channels into 

178 num_channels (int): number of channels expected in input 

179 eps: a value added to the denominator for numerical stability. Default: 1e-5 

180 affine: a boolean value that when set to ``True``, this module 

181 has learnable per-channel affine parameters initialized to ones (for weights) 

182 and zeros (for biases). Default: ``True``. 

183 

184 Shape: 

185 - Input: :math:`(N, C, *)` where :math:`C=\text{num\_channels}` 

186 - Output: :math:`(N, C, *)` (same shape as input) 

187 

188 Examples:: 

189 

190 >>> input = torch.randn(20, 6, 10, 10) 

191 >>> # Separate 6 channels into 3 groups 

192 >>> m = nn.GroupNorm(3, 6) 

193 >>> # Separate 6 channels into 6 groups (equivalent with InstanceNorm) 

194 >>> m = nn.GroupNorm(6, 6) 

195 >>> # Put all 6 channels into a single group (equivalent with LayerNorm) 

196 >>> m = nn.GroupNorm(1, 6) 

197 >>> # Activating the module 

198 >>> output = m(input) 

199 

200 .. _`Group Normalization`: https://arxiv.org/abs/1803.08494 

201 """ 

202 __constants__ = ['num_groups', 'num_channels', 'eps', 'affine'] 

203 

204 def __init__(self, num_groups, num_channels, eps=1e-5, affine=True): 

205 super(GroupNorm, self).__init__() 

206 self.num_groups = num_groups 

207 self.num_channels = num_channels 

208 self.eps = eps 

209 self.affine = affine 

210 if self.affine: 

211 self.weight = Parameter(torch.Tensor(num_channels)) 

212 self.bias = Parameter(torch.Tensor(num_channels)) 

213 else: 

214 self.register_parameter('weight', None) 

215 self.register_parameter('bias', None) 

216 self.reset_parameters() 

217 

218 def reset_parameters(self): 

219 if self.affine: 

220 init.ones_(self.weight) 

221 init.zeros_(self.bias) 

222 

223 def forward(self, input): 

224 return F.group_norm( 

225 input, self.num_groups, self.weight, self.bias, self.eps) 

226 

227 def extra_repr(self): 

228 return '{num_groups}, {num_channels}, eps={eps}, ' \ 

229 'affine={affine}'.format(**self.__dict__)
