32 # python reference of encoder 

33 def quantization_ref(X): 

34 in_shape = X.shape 

35 data_per_byte = 8 // bitwidth_ 

36 output_cols = 10 + in_shape[1] // data_per_byte 

37 tail = 0 

38 if in_shape[1] % data_per_byte: 

39 output_cols += 1 

40 tail = data_per_byte - in_shape[1] % data_per_byte 

41 segment = output_cols - 10 

42 out = np.zeros((in_shape[0], output_cols), dtype=np.uint8) 

43 for r in range(in_shape[0]): 

44 out[r][0] = bitwidth_ 

45 out[r][1] = tail 

46 min_fval = np.amin(X[r]) 

47 max_fval = np.amax(X[r]) 

48 min_bvals = bytearray(struct.pack("f", min_fval)) 

49 max_bvals = bytearray(struct.pack("f", max_fval)) 

50 for idx in range(4): 

51 out[r][2 + idx] = min_bvals[idx] 

52 out[r][6 + idx] = max_bvals[idx] 

53 for c in range(in_shape[1]): 

54 fval = X[r][c] 

55 gap = (max_fval - min_fval) / ((1 << bitwidth_) - 1) 

56 thetimes = (fval - min_fval) / (gap + 1e-8) 

57 thetimes = np.around(thetimes).astype(np.uint8) 

58 out[r][10 + c % segment] += \ 

59 (thetimes << (bitwidth_ * (c // segment))) 

60 print("pY:\n{}\n".format(out)) 

61 return (out,)

63 # the maximum quantization error 

64 def get_allowed_errors(X): 

65 out = np.zeros_like(X) 

66 for r in range(X.shape[0]): 

67 min_fval = np.amin(X[r]) 

68 max_fval = np.amax(X[r]) 

69 gap = (max_fval - min_fval) / (2 ** bitwidth_ - 1) + 1e-8 

70 out[r] = gap 

71 return out

74 def dec_ref(Y): 

75 in_shape = Y.shape 

76 bitwidth = Y[0][0] 

77 tail = Y[0][1] 

78 mask = np.array([(1 << bitwidth) - 1]).astype(np.uint8)[0] 

79 data_per_byte = 8 // bitwidth 

80 output_cols = (in_shape[1] - 10) * data_per_byte - tail 

81 segment = in_shape[1] - 10 

82 out = np.zeros((in_shape[0], output_cols), dtype=np.float) 

83 for r in range(in_shape[0]): 

84 min_fval = struct.unpack('f', Y[r][2:6])[0] 

85 max_fval = struct.unpack('f', Y[r][6:10])[0] 

86 print(min_fval, max_fval) 

87 gap = (max_fval - min_fval) / (2 ** bitwidth - 1.) + 1e-8 

88 for out_c in range(output_cols): 

89 bit_start = (out_c // segment) * bitwidth 

90 out[r][out_c] = min_fval + gap * \ 

91 ((Y[r][10 + out_c % segment] >> bit_start) & mask) 

92 print("pdecX:\n{}\n".format(out)) 

93 return (out,)
