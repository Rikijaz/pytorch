136def typename(o): 

137 if isinstance(o, torch.Tensor): 

138 return o.type() 

139 

140 module = '' 

141 class_name = '' 

142 if hasattr(o, '__module__') and o.__module__ != 'builtins' \ 

143 and o.__module__ != '__builtin__' and o.__module__ is not None: 

144 module = o.__module__ + '.' 

145 

146 if hasattr(o, '__qualname__'): 

147 class_name = o.__qualname__ 

148 elif hasattr(o, '__name__'): 

149 class_name = o.__name__ 

150 else: 

151 class_name = o.__class__.__name__ 

152 

153 return module + class_name
