117def shell(command, cwd=None, env=None): 

118 sys.stdout.flush() 

119 sys.stderr.flush() 

120 # The following cool snippet is copied from Py3 core library subprocess.call 

121 # only the with 

122 # 1. `except KeyboardInterrupt` block added for SIGINT handling. 

123 # 2. In Py2, subprocess.Popen doesn't return a context manager, so we do 

124 # `p.wait()` in a `final` block for the code to be portable. 

125 # 

126 # https://github.com/python/cpython/blob/71b6c1af727fbe13525fb734568057d78cea33f3/Lib/subprocess.py#L309-L323 

127 assert not isinstance(command, torch._six.string_classes), "Command to shell should be a list or tuple of tokens" 

128 p = subprocess.Popen(command, universal_newlines=True, cwd=cwd, env=env) 

129 try: 

130 return p.wait() 

131 except KeyboardInterrupt: 

132 # Give `p` a chance to handle KeyboardInterrupt. Without this, 

133 # `pytest` can't print errors it collected so far upon KeyboardInterrupt. 

134 exit_status = p.wait(timeout=5) 

135 if exit_status is not None: 

136 return exit_status 

137 else: 

138 p.kill() 

139 raise 

140 except: # noqa E722, copied from python core library 

141 p.kill() 

142 raise 

143 finally: 

144 # Always call p.wait() to ensure exit 

145 p.wait()

705 def genSparseTensor(self, size, sparse_dim, nnz, is_uncoalesced, device='cpu'): 

706 # Assert not given impossible combination, where the sparse dims have 

707 # empty numel, but nnz > 0 makes the indices containing values. 

708 assert all(size[d] > 0 for d in range(sparse_dim)) or nnz == 0, 'invalid arguments' 

709 

710 v_size = [nnz] + list(size[sparse_dim:]) 

711 v = torch.randn(*v_size, device=device) 

712 i = torch.rand(sparse_dim, nnz, device=device) 

713 i.mul_(torch.tensor(size[:sparse_dim]).unsqueeze(1).to(i)) 

714 i = i.to(torch.long) 

715 if is_uncoalesced: 

716 v = torch.cat([v, torch.randn_like(v)], 0) 

717 i = torch.cat([i, i], 1) 

718 

719 x = torch.sparse_coo_tensor(i, v, torch.Size(size)) 

720 

721 if not is_uncoalesced: 

722 x = x.coalesce() 

723 else: 

724 # FIXME: `x` is a sparse view of `v`. Currently rebase_history for 

725 # sparse views is not implemented, so this workaround is 

726 # needed for inplace operations done on `x`, e.g., copy_(). 

727 # Remove after implementing something equivalent to CopySlice 

728 # for sparse views. 

729 # NOTE: We do clone() after detach() here because we need to be able to change size/storage of x afterwards 

730 x = x.detach().clone() 

731 return x, x._indices().clone(), x._values().clone()

737 def safeCoalesce(self, t): 

738 tc = t.coalesce() 

739 self.assertEqual(tc.to_dense(), t.to_dense()) 

740 self.assertTrue(tc.is_coalesced()) 

741 

742 # Our code below doesn't work when nnz is 0, because 

743 # then it's a 0D tensor, not a 2D tensor. 

744 if t._nnz() == 0: 

745 self.assertEqual(t._indices(), tc._indices()) 

746 self.assertEqual(t._values(), tc._values()) 

747 return tc 

748 

749 value_map = {} 

750 for idx, val in zip(t._indices().t(), t._values()): 

751 idx_tup = tuple(idx.tolist()) 

752 if idx_tup in value_map: 

753 value_map[idx_tup] += val 

754 else: 

755 value_map[idx_tup] = val.clone() if isinstance(val, torch.Tensor) else val 

756 

757 new_indices = sorted(list(value_map.keys())) 

758 new_values = [value_map[idx] for idx in new_indices] 

759 if t._values().ndimension() < 2: 

760 new_values = t._values().new(new_values) 

761 else: 

762 new_values = torch.stack(new_values) 

763 

764 new_indices = t._indices().new(new_indices).t() 

765 tg = t.new(new_indices, new_values, t.size()) 

766 

767 self.assertEqual(tc._indices(), tg._indices()) 

768 self.assertEqual(tc._values(), tg._values()) 

769 

770 if t.is_coalesced(): 

771 self.assertEqual(tc._indices(), t._indices()) 

772 self.assertEqual(tc._values(), t._values()) 

773 

774 return tg

796 def assertTensorsEqual(a, b): 

797 super(TestCase, self).assertEqual(a.size(), b.size(), message) 

798 if a.numel() > 0: 

799 if (a.device.type == 'cpu' and (a.dtype == torch.float16 or a.dtype == torch.bfloat16)): 

800 # CPU half and bfloat16 tensors don't have the methods we need below 

801 a = a.to(torch.float32) 

802 if (a.device.type == 'cuda' and a.dtype == torch.bfloat16): 

803 # CUDA bfloat16 tensors don't have the methods we need below 

804 a = a.to(torch.float32) 

805 b = b.to(a) 

806 

807 if (a.dtype == torch.bool) != (b.dtype == torch.bool): 

808 raise TypeError("Was expecting both tensors to be bool type.") 

809 else: 

810 if a.dtype == torch.bool and b.dtype == torch.bool: 

811 # we want to respect precision but as bool doesn't support subtraction, 

812 # boolean tensor has to be converted to int 

813 a = a.to(torch.int) 

814 b = b.to(torch.int) 

815 

816 diff = a - b 

817 if a.is_floating_point(): 

818 # check that NaNs are in the same locations 

819 nan_mask = torch.isnan(a) 

820 self.assertTrue(torch.equal(nan_mask, torch.isnan(b)), message) 

821 diff[nan_mask] = 0 

822 # inf check if allow_inf=True 

823 if allow_inf: 

824 inf_mask = torch.isinf(a) 

825 inf_sign = inf_mask.sign() 

826 self.assertTrue(torch.equal(inf_sign, torch.isinf(b).sign()), message) 

827 diff[inf_mask] = 0 

828 # TODO: implement abs on CharTensor (int8) 

829 if diff.is_signed() and diff.dtype != torch.int8: 

830 diff = diff.abs() 

831 max_err = diff.max() 

832 self.assertLessEqual(max_err, prec, message) 

833 super(TestCase, self).assertEqual(x.is_sparse, y.is_sparse, message) 

834 super(TestCase, self).assertEqual(x.is_quantized, y.is_quantized, message) 

835 if x.is_sparse: 

836 x = self.safeCoalesce(x) 

837 y = self.safeCoalesce(y) 

838 assertTensorsEqual(x._indices(), y._indices()) 

839 assertTensorsEqual(x._values(), y._values()) 

840 elif x.is_quantized and y.is_quantized: 

841 self.assertEqual(x.qscheme(), y.qscheme(), prec=prec, 

842 message=message, allow_inf=allow_inf)

844 self.assertEqual(x.q_scale(), y.q_scale(), prec=prec, 

845 message=message, allow_inf=allow_inf) 

846 self.assertEqual(x.q_zero_point(), y.q_zero_point(), 

847 prec=prec, message=message, 

848 allow_inf=allow_inf) 

849 elif x.qscheme() == torch.per_channel_affine: 

850 self.assertEqual(x.q_per_channel_scales(), y.q_per_channel_scales(), prec=prec, 

851 message=message, allow_inf=allow_inf) 

852 self.assertEqual(x.q_per_channel_zero_points(), y.q_per_channel_zero_points(), 

853 prec=prec, message=message, 

854 allow_inf=allow_inf) 

855 self.assertEqual(x.q_per_channel_axis(), y.q_per_channel_axis(), 

856 prec=prec, message=message) 

857 self.assertEqual(x.dtype, y.dtype) 

858 self.assertEqual(x.int_repr().to(torch.int32), 

859 y.int_repr().to(torch.int32), prec=prec, 

860 message=message, allow_inf=allow_inf) 

861 else: 

862 assertTensorsEqual(x, y) 

863 elif isinstance(x, string_classes) and isinstance(y, string_classes): 

864 super(TestCase, self).assertEqual(x, y, message) 

865 elif type(x) == set and type(y) == set: 

866 super(TestCase, self).assertEqual(x, y, message) 

867 elif isinstance(x, dict) and isinstance(y, dict): 

868 if isinstance(x, OrderedDict) and isinstance(y, OrderedDict): 

869 self.assertEqual(x.items(), y.items(), prec=prec, 

870 message=message, allow_inf=allow_inf) 

871 else: 

872 self.assertEqual(set(x.keys()), set(y.keys()), prec=prec, 

873 message=message, allow_inf=allow_inf) 

874 key_list = list(x.keys()) 

875 self.assertEqual([x[k] for k in key_list], 

876 [y[k] for k in key_list], 

877 prec=prec, message=message, 

878 allow_inf=allow_inf) 

879 elif is_iterable(x) and is_iterable(y): 

880 super(TestCase, self).assertEqual(len(x), len(y), message) 

881 for x_, y_ in zip(x, y): 

882 self.assertEqual(x_, y_, prec=prec, message=message, 

883 allow_inf=allow_inf) 

884 elif isinstance(x, bool) and isinstance(y, bool): 

885 super(TestCase, self).assertEqual(x, y, message) 

886 elif isinstance(x, Number) and isinstance(y, Number): 

887 if abs(x) == inf or abs(y) == inf: 

888 if allow_inf: 

889 super(TestCase, self).assertEqual(x, y, message) 

890 else: 

891 self.fail("Expected finite numeric values - x={}, y={}".format(x, y)) 

892 return 

893 super(TestCase, self).assertLessEqual(abs(x - y), prec, message) 

894 else: 

895 super(TestCase, self).assertEqual(x, y, message)

1308def brute_pdist(inp, p=2): 

1309 """Computes the same as torch.pdist using primitives""" 

1310 n = inp.shape[-2] 

1311 k = n * (n - 1) // 2 

1312 if k == 0: 

1313 # torch complains about empty indices 

1314 return torch.empty(inp.shape[:-2] + (0,), dtype=inp.dtype, device=inp.device) 

1315 square = torch.norm(inp[..., None, :] - inp[..., None, :, :], p=p, dim=-1) 

1316 unroll = square.view(square.shape[:-2] + (n * n,)) 

1317 inds = torch.ones(k, dtype=torch.int) 

1318 inds[torch.arange(n - 1, 1, -1, dtype=torch.int).cumsum(0)] += torch.arange(2, n, dtype=torch.int) 

1319 return unroll[..., inds.cumsum(0)]
