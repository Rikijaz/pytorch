29def dist_init(old_test_method=None, setup_rpc=True, clean_shutdown=True): 

30 """ 

31 We use this decorator for setting up and tearing down state since 

32 MultiProcessTestCase runs each `test*` method in a separate process and 

33 each process just runs the `test*` method without actually calling 

34 'setUp' and 'tearDown' methods of unittest. 

35 """ 

36 

37 # If we use dist_init without arguments (ex: @dist_init), old_test_method is 

38 # appropriately set and we return the wrapper appropriately. On the other 

39 # hand if dist_init has arguments (ex: @dist_init(clean_shutdown=False)), 

40 # old_test_method is None and we return a functools.partial which is the real 

41 # decorator that is used and as a result we recursively call dist_init with 

42 # old_test_method and the rest of the arguments appropriately set. 

43 if old_test_method is None: 

44 return partial(

114def wait_until_pending_users_flushed(): 

115 ''' 

116 The RRef protocol holds forkIds of rrefs in a map until those forks are 

117 confirmed by the owner. The message confirming the fork may arrive after 

118 our tests check whether this map is empty, which leads to failures and 

119 flaky tests. to_here also does not guarantee that we have finished 

120 processind the owner's confirmation message for the RRef. This function 

121 loops until the map is empty, which means the messages have been received 

122 as processed. Call this function before asserting the map returned by 

123 _get_debug_info is empty. 

124 ''' 

125 num_pending_users = int(_rref_context_get_debug_info()["num_pending_users"])

127 time.sleep(0.1)
