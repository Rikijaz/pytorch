463def get_expected_file(function_ptr): 

464 module_id = function_ptr.__class__.__module__ 

465 test_file = sys.modules[module_id].__file__ 

466 # Look for the .py file (since __file__ could be pyc). 

467 test_file = ".".join(test_file.split('.')[:-1]) + '.py' 

468 

469 # Use realpath to follow symlinks appropriately. 

470 test_dir = os.path.dirname(os.path.realpath(test_file)) 

471 functionName = function_ptr.id().split('.')[-1] 

472 return os.path.join(test_dir, 

473 "expect", 

474 'TestTensorBoard.' + functionName + ".expect")

482def compare_image_proto(actual_proto, function_ptr): 

483 expected_str = read_expected_content(function_ptr) 

484 expected_proto = Summary() 

485 text_format.Parse(expected_str, expected_proto) 

486 

487 [actual, expected] = [actual_proto.value[0], expected_proto.value[0]] 

488 actual_img = Image.open(io.BytesIO(actual.image.encoded_image_string)) 

489 expected_img = Image.open(io.BytesIO(expected.image.encoded_image_string)) 

490 

491 return ( 

492 actual.tag == expected.tag and 

493 actual.image.height == expected.image.height and 

494 actual.image.width == expected.image.width and 

495 actual.image.colorspace == expected.image.colorspace and 

496 actual_img == expected_img 

497 )
